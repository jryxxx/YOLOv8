  7. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  8.     # Pad to 'same' shape outputs
  9.     if d > 1:
  10.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  11.     if p is None:
  12.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  13.     return p

  16. class Conv(nn.Module):
  17.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  18.     default_act = nn.SiLU()  # default activation
  19.  
  20.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  21.         super().__init__()
  22.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  23.         self.bn = nn.BatchNorm2d(c2)
  24.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  25.  
  26.     def forward(self, x):
  27.         return self.act(self.bn(self.conv(x)))
  28.  
  29.     def forward_fuse(self, x):
  30.         return self.act(self.conv(x))
  31.  
  32.  
  33.  
  34. class Bottleneck(nn.Module):
  35.     # Standard bottleneck
  36.     def __init__(self, c1, c2, shortcut=True, g=1, k=(3, 3), e=0.5):  # ch_in, ch_out, shortcut, groups, kernels, expand
  37.         super().__init__()
  38.         c_ = int(c2 * e)  # hidden channels
  39.         self.cv1 = Conv(c1, c_, k[0], 1)
  40.         self.cv2 = Conv(c_, c2, k[1], 1, g=g)
  41.         self.add = shortcut and c1 == c2
  42.  
  43.     def forward(self, x):
  44.         return x + self.cv2(self.cv1(x)) if self.add else self.cv2(self.cv1(x))
  45.  
  46. class C2f(nn.Module):
  47.     # CSP Bottleneck with 2 convolutions
  48.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  49.         super().__init__()
  50.         self.c = int(c2 * e)  # hidden channels
  51.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  52.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  53.         self.m = nn.ModuleList(Bottleneck(self.c, self.c, shortcut, g, k=((3, 3), (3, 3)), e=1.0) for _ in range(n))
  54.  
  55.     def forward(self, x):
  56.         y = list(self.cv1(x).chunk(2, 1))
  57.         y.extend(m(y[-1]) for m in self.m)
  58.         return self.cv2(torch.cat(y, 1))
  59.  
  60.     def forward_split(self, x):
  61.         y = list(self.cv1(x).split((self.c, self.c), 1))
  62.         y.extend(m(y[-1]) for m in self.m)
  63.         return self.cv2(torch.cat(y, 1))

  65. class MSBlockLayer(nn.Module):
  66.     def __init__(self, inc, ouc, k) -> None:
  67.         super().__init__()
  68.  
  69.         self.in_conv = Conv(inc, ouc, 1)
  70.         self.mid_conv = Conv(ouc, ouc, k, g=ouc)
  71.         self.out_conv = Conv(ouc, inc, 1)
  72.  
  73.     def forward(self, x):
  74.         return self.out_conv(self.mid_conv(self.in_conv(x)))
  75.

  77. class MSBlock(nn.Module):
  78.     def __init__(self, inc, ouc, kernel_sizes, in_expand_ratio=3., mid_expand_ratio=2., layers_num=3,
  79.                  in_down_ratio=2.) -> None:
  80.         super().__init__()
  81.  
  82.         in_channel = int(inc * in_expand_ratio // in_down_ratio)
  83.         self.mid_channel = in_channel // len(kernel_sizes)
  84.         groups = int(self.mid_channel * mid_expand_ratio)
  85.         self.in_conv = Conv(inc, in_channel)
  86.  
  87.         self.mid_convs = []
  88.         for kernel_size in kernel_sizes:
  89.             if kernel_size == 1:
  90.                 self.mid_convs.append(nn.Identity())
  91.                 continue
  92.             mid_convs = [MSBlockLayer(self.mid_channel, groups, k=kernel_size) for _ in range(int(layers_num))]
  93.             self.mid_convs.append(nn.Sequential(*mid_convs))
  94.         self.mid_convs = nn.ModuleList(self.mid_convs)
  95.         self.out_conv = Conv(in_channel, ouc, 1)
  96.  
  97.         self.attention = None
  98.  
  99.     def forward(self, x):
  100.         out = self.in_conv(x)
  101.         channels = []
  102.         for i, mid_conv in enumerate(self.mid_convs):
  103.             channel = out[:, i * self.mid_channel:(i + 1) * self.mid_channel, ...]
  104.             if i >= 1:
  105.                 channel = channel + channels[i - 1]
  106.             channel = mid_conv(channel)
  107.             channels.append(channel)
  108.         out = torch.cat(channels, dim=1)
  109.         out = self.out_conv(out)
  110.         if self.attention is not None:
  111.             out = self.attention(out)
  112.         return out

1.DCN V2介绍

DCN V2: Improved Deep & Cross Network and Practical Lessons for Web-scale Learning to Rank Systems

论文：https://arxiv.org/abs/2008.13535

 

作者通过在DCN的基础上，增加了2个创新点，分别是调制模块和使用多个调制后的DCN模块，从形成了DCN的升级版本——DCN-v2!
①调制模块：
除了学习偏移参数Δ p \Delta pΔp(offset)之外，还要通过调制学习一个变化幅度Δ m \Delta mΔm。通过这个幅度来进一步合理控制新采样点的偏移范围。经过调制后的单个DCN我们记为mDCN(modulated-DCN)。

②多个调制DCN的堆积：
通过堆积多个调制mDCN来增加offset的偏移范围，同时显然多个块的堆积对偏移的精确性也是有一定校正细化的作用的，即进一步增强其对抗空间变化的能力(coarse-to-fine)。

为了验证DCN-v2相对DCN-v1、常规CNN对表现力的提升作用，作者在COCO 2017数据集上，用可变形卷积替换网络模型中的第3~5层中的卷积层，实验结果如下：

  1. ###################### DCNV2  ####     start#############################
  2.  
  3.  
  4. class DCN(nn.Module):
  5.     def __init__(self, in_channels, out_channels, kernel_size, stride=1,
  6.                  padding=1, dilation=1, groups=1, deformable_groups=1):
  7.         super(DCN, self).__init__()
  8.  
  9.         self.in_channels = in_channels
  10.         self.out_channels = out_channels
  11.         self.kernel_size = (kernel_size, kernel_size)
  12.         self.stride = (stride, stride)
  13.         self.padding = (padding, padding)
  14.         self.dilation = (dilation, dilation)
  15.         self.groups = groups
  16.         self.deformable_groups = deformable_groups
  17.  
  18.         self.weight = nn.Parameter(
  19.             torch.empty(out_channels, in_channels, *self.kernel_size)
  20.         )
  21.         self.bias = nn.Parameter(torch.empty(out_channels))
  22.  
  23.         out_channels_offset_mask = (self.deformable_groups * 3 *
  24.                                     self.kernel_size[0] * self.kernel_size[1])
  25.         self.conv_offset_mask = nn.Conv2d(
  26.             self.in_channels,
  27.             out_channels_offset_mask,
  28.             kernel_size=self.kernel_size,
  29.             stride=self.stride,
  30.             padding=self.padding,
  31.             bias=True,
  32.         )
  33.         self.bn = nn.BatchNorm2d(out_channels)
  34.         self.act = Conv.default_act
  35.         self.reset_parameters()
  36.  
  37.     def forward(self, x):
  38.         offset_mask = self.conv_offset_mask(x)
  39.         o1, o2, mask = torch.chunk(offset_mask, 3, dim=1)
  40.         offset = torch.cat((o1, o2), dim=1)
  41.         mask = torch.sigmoid(mask)
  42.         x = torch.ops.torchvision.deform_conv2d(
  43.             x,
  44.             self.weight,
  45.             offset,
  46.             mask,
  47.             self.bias,
  48.             self.stride[0], self.stride[1],
  49.             self.padding[0], self.padding[1],
  50.             self.dilation[0], self.dilation[1],
  51.             self.groups,
  52.             self.deformable_groups,
  53.             True
  54.         )
  55.         x = self.bn(x)
  56.         x = self.act(x)
  57.         return x
  58.  
  59.     def reset_parameters(self):
  60.         n = self.in_channels
  61.         for k in self.kernel_size:
  62.             n *= k
  63.         std = 1. / math.sqrt(n)
  64.         self.weight.data.uniform_(-std, std)
  65.         self.bias.data.zero_()
  66.         self.conv_offset_mask.weight.data.zero_()
  67.         self.conv_offset_mask.bias.data.zero_()
  68.  
  69. class Bottleneck_DCN(nn.Module):
  70.     # Standard bottleneck with DCN
  71.     def __init__(self, c1, c2, shortcut=True, g=1, k=(3, 3), e=0.5):  # ch_in, ch_out, shortcut, groups, kernels, expand
  72.         super().__init__()
  73.         c_ = int(c2 * e)  # hidden channels
  74.         if k[0] == 3:
  75.             self.cv1 = DCN(c1, c_, k[0], 1)
  76.         else:
  77.             self.cv1 = Conv(c1, c_, k[0], 1)
  78.         if k[1] == 3:
  79.             self.cv2 = DCN(c_, c2, k[1], 1, groups=g)
  80.         else:
  81.             self.cv2 = Conv(c_, c2, k[1], 1, g=g)
  82.         self.add = shortcut and c1 == c2
  83.  
  84.     def forward(self, x):
  85.         return x + self.cv2(self.cv1(x)) if self.add else self.cv2(self.cv1(x))
  86.  
  87. class DCNV2(nn.Module):
  88.     # CSP Bottleneck with 2 convolutions
  89.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  90.         super().__init__()
  91.         self.c = int(c2 * e)  # hidden channels
  92.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  93.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  94.         self.m = nn.ModuleList(Bottleneck_DCN(self.c, self.c, shortcut, g, k=(3, 3), e=1.0) for _ in range(n))
  95.  
  96.     def forward(self, x):
  97.         y = list(self.cv1(x).split((self.c, self.c), 1))
  98.         y.extend(m(y[-1]) for m in self.m)
  99.         return self.cv2(torch.cat(y, 1))
  100.  
  101. ###################### DCNV2  ####     end#############################

  1. ###################### DCNV3  ####     start#############################

  3. from ultralytics.nn.ops_dcnv3.modules import DCNv3
  4.  
  5. class DCNV3_YoLo(nn.Module):
  6.     def __init__(self, inc, ouc, k=1, s=1, p=None, g=1, d=1, act=True):
  7.         super().__init__()
  8.  
  9.         self.conv = Conv(inc, ouc, k=1)
  10.         self.dcnv3 = DCNv3(ouc, kernel_size=k, stride=s, group=g, dilation=d)
  11.         self.bn = nn.BatchNorm2d(ouc)
  12.         self.act = Conv.default_act
  13.  
  14.     def forward(self, x):
  15.         x = self.conv(x)
  16.         x = x.permute(0, 2, 3, 1)
  17.         x = self.dcnv3(x)
  18.         x = x.permute(0, 3, 1, 2)
  19.         x = self.act(self.bn(x))
  20.         return x
  21.  
  22. class Bottleneck_DCNV3(nn.Module):
  23.     # Standard bottleneck
  24.     def __init__(self, c1, c2, shortcut=True, g=1, k=(3, 3), e=0.5):  # ch_in, ch_out, shortcut, groups, kernels, expand
  25.         super().__init__()
  26.         c_ = int(c2 * e)  # hidden channels
  27.         self.cv1 = Conv(c1, c_, k[0], 1)
  28.         self.cv2 = DCNV3_YoLo(c_, c2, k[1], 1, g=g)
  29.         self.add = shortcut and c1 == c2
  30.  
  31.     def forward(self, x):
  32.         return x + self.cv2(self.cv1(x)) if self.add else self.cv2(self.cv1(x))
  33.  
  34. class DCNV3(nn.Module):
  35.     # CSP Bottleneck with 2 convolutions
  36.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  37.         super().__init__()
  38.         self.c = int(c2 * e)  # hidden channels
  39.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  40.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  41.         self.m = nn.ModuleList(Bottleneck_DCNV3(self.c, self.c, shortcut, g, k=(3, 3), e=1.0) for _ in range(n))
  42.  
  43.     def forward(self, x):
  44.         y = list(self.cv1(x).chunk(2, 1))
  45.         y.extend(m(y[-1]) for m in self.m)
  46.         return self.cv2(torch.cat(y, 1))
  47.  
  48.     def forward_split(self, x):
  49.         y = list(self.cv1(x).split((self.c, self.c), 1))
  50.         y.extend(m(y[-1]) for m in self.m)
  51.         return self.cv2(torch.cat(y, 1))

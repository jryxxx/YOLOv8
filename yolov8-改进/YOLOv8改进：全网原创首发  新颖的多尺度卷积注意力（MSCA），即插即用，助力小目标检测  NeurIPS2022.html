  6. class MSCAAttention(nn.Module):
  7.     # SegNext NeurIPS 2022
  8.     # https://github.com/Visual-Attention-Network/SegNeXt/tree/main
  9.     def __init__(self, dim):
  10.         super().__init__()
  11.         self.conv0 = nn.Conv2d(dim, dim, 5, padding=2, groups=dim)
  12.         self.conv0_1 = nn.Conv2d(dim, dim, (1, 7), padding=(0, 3), groups=dim)
  13.         self.conv0_2 = nn.Conv2d(dim, dim, (7, 1), padding=(3, 0), groups=dim)
  14.  
  15.         self.conv1_1 = nn.Conv2d(dim, dim, (1, 11), padding=(0, 5), groups=dim)
  16.         self.conv1_2 = nn.Conv2d(dim, dim, (11, 1), padding=(5, 0), groups=dim)
  17.  
  18.         self.conv2_1 = nn.Conv2d(dim, dim, (1, 21), padding=(0, 10), groups=dim)
  19.         self.conv2_2 = nn.Conv2d(dim, dim, (21, 1), padding=(10, 0), groups=dim)
  20.         self.conv3 = nn.Conv2d(dim, dim, 1)
  21.  
  22.     def forward(self, x):
  23.         u = x.clone()
  24.         attn = self.conv0(x)
  25.  
  26.         attn_0 = self.conv0_1(attn)
  27.         attn_0 = self.conv0_2(attn_0)
  28.  
  29.         attn_1 = self.conv1_1(attn)
  30.         attn_1 = self.conv1_2(attn_1)
  31.  
  32.         attn_2 = self.conv2_1(attn)
  33.         attn_2 = self.conv2_2(attn_2)
  34.         attn = attn + attn_0 + attn_1 + attn_2
  35.  
  36.         attn = self.conv3(attn)
  37.  
  38.         return attn * u

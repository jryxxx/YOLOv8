 2.Ghostnet介绍

论文： https://arxiv.org/pdf/1911.11907.pdf

       摘要： 由于内存和计算资源的限制，在嵌入式设备商部署卷积神经网络CNN很困难。Deploying convolutional neural networks （CNNs）on embedded devices is difficulty due to the limited memory and computation resources. 特征图中的冗余是成功神经网络的重要特征，但在神经网络架构设计中很少研究，本文提出一种新的Ghost模块，从廉价的操作中生成更多的特征图。基于一组内在的特征图，以低成本应用一系列线性变换来生成许多ghost特征图，可以充分揭示内在特征的信息。所提出的Ghost模块可以作为即插即用的组件轻松升级现有的卷积神经网络，Ghost bottleneck被设计为堆叠Ghost模块 ，然后轻松建立轻量级GhostNet。

        在主流的深度神经网络提取的特征图中，丰富甚至冗余的信息通常保证了对输入数据的全面理解。例如，图1展示了由ResNet-50生成的输入图像的一些特征图，并且存在许多相似的特征图对，就像彼此之间的幽灵一样

        引入了一个新的 Ghost 模块，通过使用更少的参数来生成更多的特征。具体来说，深度神经网络中的一个普通卷积层会被分成两部分。第一部分涉及普通卷积，但它们的总数将受到严格控制。给定第一部分的内在特征图，然后应用一系列简单的线性操作来生成更多的特征图。在不改变输出特征图的大小的情况下，与普通卷积神经网络相比，这个 Ghost 模块所需的总体参数数量和计算复杂度有所降低。

        基于 Ghost 模块，建立了一个高效的神经架构，即 GhostNet。首先替换基准神经架构中的原始卷积层以证明 Ghost 模块的有效性，然后验证 GhostNets 在几个基准视觉数据集上的优越性。

 

         利用ghost模块的优势，介绍了专门为小模型设计Ghost bottlenck，如图3所示。Ghost bottleneck类似于ResNet中基本残差块，其中集成了几个卷积层和shortcuts。ghost bottleneck主要有堆叠的ghost module组成，第一个ghost模块充当增加通道数量的扩展层，将输出通道数和输入通道数之比成为扩展比。第二个ghost模块减少通道数量以匹配shortcuts，然后，shortcuts在这两个ghost模块的输入和输出之间。BN和ReLu在每一层之后应用，第二个ghost模块后不用ReLU。

        实验结果表明，所提出的 Ghost 模块能够降低通用卷积层的计算成本，同时保持相似的识别性能，并且 GhostNets 可以在各种任务上超越SOTA高效深度模型，如 MobileNetV3 移动设备上的快速推理。

  3. """
  4. Creates a GhostNet Model as defined in:
  5. GhostNet: More Features from Cheap Operations By Kai Han, Yunhe Wang, Qi Tian, Jianyuan Guo, Chunjing Xu, Chang Xu.
  6. https://arxiv.org/abs/1911.11907
  7. Modified from https://github.com/d-li14/mobilenetv3.pytorch and https://github.com/rwightman/pytorch-image-models
  8. """
  9. import torch
  10. import torch.nn as nn
  11. import torch.nn.functional as F
  12. import math
  13.  
  14. __all__ = ['ghostnet']
  15.  
  16.  
  17. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  18.     # Pad to 'same' shape outputs
  19.     if d > 1:
  20.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  21.     if p is None:
  22.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  23.     return p
  24.  
  25.  
  26. class Conv(nn.Module):
  27.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  28.     default_act = nn.SiLU()  # default activation
  29.  
  30.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  31.         super().__init__()
  32.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  33.         self.bn = nn.BatchNorm2d(c2)
  34.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  35.  
  36.     def forward(self, x):
  37.         return self.act(self.bn(self.conv(x)))
  38.  
  39.     def forward_fuse(self, x):
  40.         return self.act(self.conv(x))
  41.  
  42.  
  43. def _make_divisible(v, divisor, min_value=None):
  44.     """
  45.     This function is taken from the original tf repo.
  46.     It ensures that all layers have a channel number that is divisible by 8
  47.     It can be seen here:
  48.     https://github.com/tensorflow/models/blob/master/research/slim/nets/mobilenet/mobilenet.py
  49.     """
  50.     if min_value is None:
  51.         min_value = divisor
  52.     new_v = max(min_value, int(v + divisor / 2) // divisor * divisor)
  53.     # Make sure that round down does not go down by more than 10%.
  54.     if new_v < 0.9 * v:
  55.         new_v += divisor
  56.     return new_v
  57.  
  58.  
  59. def hard_sigmoid(x, inplace: bool = False):
  60.     if inplace:
  61.         return x.add_(3.).clamp_(0., 6.).div_(6.)
  62.     else:
  63.         return F.relu6(x + 3.) / 6.
  64.  
  65.  
  66. class SqueezeExcite(nn.Module):
  67.     def __init__(self, in_chs, se_ratio=0.25, reduced_base_chs=None,
  68.                  act_layer=nn.ReLU, gate_fn=hard_sigmoid, divisor=4, **_):
  69.         super(SqueezeExcite, self).__init__()
  70.         self.gate_fn = gate_fn
  71.         reduced_chs = _make_divisible((reduced_base_chs or in_chs) * se_ratio, divisor)
  72.         self.avg_pool = nn.AdaptiveAvgPool2d(1)
  73.         self.conv_reduce = nn.Conv2d(in_chs, reduced_chs, 1, bias=True)
  74.         self.act1 = act_layer(inplace=True)
  75.         self.conv_expand = nn.Conv2d(reduced_chs, in_chs, 1, bias=True)
  76.  
  77.     def forward(self, x):
  78.         x_se = self.avg_pool(x)
  79.         x_se = self.conv_reduce(x_se)
  80.         x_se = self.act1(x_se)
  81.         x_se = self.conv_expand(x_se)
  82.         x = x * self.gate_fn(x_se)
  83.         return x
  84.  
  85.  
  86. class ConvBnAct(nn.Module):
  87.     def __init__(self, in_chs, out_chs, kernel_size,
  88.                  stride=1, act_layer=nn.ReLU):
  89.         super(ConvBnAct, self).__init__()
  90.         self.conv = nn.Conv2d(in_chs, out_chs, kernel_size, stride, kernel_size // 2, bias=False)
  91.         self.bn1 = nn.BatchNorm2d(out_chs)
  92.         self.act1 = act_layer(inplace=True)
  93.  
  94.     def forward(self, x):
  95.         x = self.conv(x)
  96.         x = self.bn1(x)
  97.         x = self.act1(x)
  98.         return x
  99.  
  100.  
  101. class GhostModule(nn.Module):
  102.     def __init__(self, inp, oup, kernel_size=1, ratio=2, dw_size=3, stride=1, relu=True):
  103.         super(GhostModule, self).__init__()
  104.         self.oup = oup
  105.         init_channels = math.ceil(oup / ratio)
  106.         new_channels = init_channels * (ratio - 1)
  107.  
  108.         self.primary_conv = nn.Sequential(
  109.             nn.Conv2d(inp, init_channels, kernel_size, stride, kernel_size // 2, bias=False),
  110.             nn.BatchNorm2d(init_channels),
  111.             nn.ReLU(inplace=True) if relu else nn.Sequential(),
  112.         )
  113.  
  114.         self.cheap_operation = nn.Sequential(
  115.             nn.Conv2d(init_channels, new_channels, dw_size, 1, dw_size // 2, groups=init_channels, bias=False),
  116.             nn.BatchNorm2d(new_channels),
  117.             nn.ReLU(inplace=True) if relu else nn.Sequential(),
  118.         )
  119.  
  120.     def forward(self, x):
  121.         x1 = self.primary_conv(x)
  122.         x2 = self.cheap_operation(x1)
  123.         out = torch.cat([x1, x2], dim=1)
  124.         return out[:, :self.oup, :, :]
  125.  
  126.  
  127. class GhostBottleneck(nn.Module):
  128.     """ Ghost bottleneck w/ optional SE"""
  129.  
  130.     def __init__(self, in_chs, mid_chs, out_chs, dw_kernel_size=3,
  131.                  stride=1, act_layer=nn.ReLU, se_ratio=0.):
  132.         super(GhostBottleneck, self).__init__()
  133.         has_se = se_ratio is not None and se_ratio > 0.
  134.         self.stride = stride
  135.  
  136.         # Point-wise expansion
  137.         self.ghost1 = GhostModule(in_chs, mid_chs, relu=True)
  138.  
  139.         # Depth-wise convolution
  140.         if self.stride > 1:
  141.             self.conv_dw = nn.Conv2d(mid_chs, mid_chs, dw_kernel_size, stride=stride,
  142.                                      padding=(dw_kernel_size - 1) // 2,
  143.                                      groups=mid_chs, bias=False)
  144.             self.bn_dw = nn.BatchNorm2d(mid_chs)
  145.  
  146.         # Squeeze-and-excitation
  147.         if has_se:
  148.             self.se = SqueezeExcite(mid_chs, se_ratio=se_ratio)
  149.         else:
  150.             self.se = None
  151.  
  152.         # Point-wise linear projection
  153.         self.ghost2 = GhostModule(mid_chs, out_chs, relu=False)
  154.  
  155.         # shortcut
  156.         if (in_chs == out_chs and self.stride == 1):
  157.             self.shortcut = nn.Sequential()
  158.         else:
  159.             self.shortcut = nn.Sequential(
  160.                 nn.Conv2d(in_chs, in_chs, dw_kernel_size, stride=stride,
  161.                           padding=(dw_kernel_size - 1) // 2, groups=in_chs, bias=False),
  162.                 nn.BatchNorm2d(in_chs),
  163.                 nn.Conv2d(in_chs, out_chs, 1, stride=1, padding=0, bias=False),
  164.                 nn.BatchNorm2d(out_chs),
  165.             )
  166.  
  167.     def forward(self, x):
  168.         residual = x
  169.  
  170.         # 1st ghost bottleneck
  171.         x = self.ghost1(x)
  172.  
  173.         # Depth-wise convolution
  174.         if self.stride > 1:
  175.             x = self.conv_dw(x)
  176.             x = self.bn_dw(x)
  177.  
  178.         # Squeeze-and-excitation
  179.         if self.se is not None:
  180.             x = self.se(x)
  181.  
  182.         # 2nd ghost bottleneck
  183.         x = self.ghost2(x)
  184.  
  185.         x += self.shortcut(residual)
  186.         return x
  187.  
  188.  
  189. class C2f_GhostBottleneck(nn.Module):
  190.     # CSP Bottleneck with 2 convolutions
  191.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  192.         super().__init__()
  193.         self.c = int(c2 * e)  # hidden channels
  194.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  195.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  196.         self.m = nn.ModuleList(GhostBottleneck(self.c, self.c, self.c) for _ in range(n))
  197.  
  198.     def forward(self, x):
  199.         y = list(self.cv1(x).chunk(2, 1))
  200.         y.extend(m(y[-1]) for m in self.m)
  201.         return self.cv2(torch.cat(y, 1))
  202.  
  203. class GhostNet(nn.Module):
  204.     def __init__(self, cfgs, num_classes=1000, width=1.0, dropout=0.2):
  205.         super(GhostNet, self).__init__()
  206.         # setting of inverted residual blocks
  207.         self.cfgs = cfgs
  208.         self.dropout = dropout
  209.  
  210.         # building first layer
  211.         output_channel = _make_divisible(16 * width, 4)
  212.         self.conv_stem = nn.Conv2d(3, output_channel, 3, 2, 1, bias=False)
  213.         self.bn1 = nn.BatchNorm2d(output_channel)
  214.         self.act1 = nn.ReLU(inplace=True)
  215.         input_channel = output_channel
  216.  
  217.         # building inverted residual blocks
  218.         stages = []
  219.         block = GhostBottleneck
  220.         for cfg in self.cfgs:
  221.             layers = []
  222.             for k, exp_size, c, se_ratio, s in cfg:
  223.                 output_channel = _make_divisible(c * width, 4)
  224.                 hidden_channel = _make_divisible(exp_size * width, 4)
  225.                 layers.append(block(input_channel, hidden_channel, output_channel, k, s,
  226.                                     se_ratio=se_ratio))
  227.                 input_channel = output_channel
  228.             stages.append(nn.Sequential(*layers))
  229.  
  230.         output_channel = _make_divisible(exp_size * width, 4)
  231.         stages.append(nn.Sequential(ConvBnAct(input_channel, output_channel, 1)))
  232.         input_channel = output_channel
  233.  
  234.         self.blocks = nn.Sequential(*stages)
  235.  
  236.         # building last several layers
  237.         output_channel = 1280
  238.         self.global_pool = nn.AdaptiveAvgPool2d((1, 1))
  239.         self.conv_head = nn.Conv2d(input_channel, output_channel, 1, 1, 0, bias=True)
  240.         self.act2 = nn.ReLU(inplace=True)
  241.         self.classifier = nn.Linear(output_channel, num_classes)
  242.         self.channel = [i.size(-1) for i in self.forward(torch.randn(1, 3, 640, 640))]
  243.  
  244.     def forward(self, x):
  245.         x = self.conv_stem(x)
  246.         x = self.bn1(x)
  247.         x = self.act1(x)
  248.         x = self.blocks(x)
  249.         x = self.global_pool(x)
  250.         x = self.conv_head(x)
  251.         x = self.act2(x)
  252.         x = x.view(x.size(0), -1)
  253.         if self.dropout > 0.:
  254.             x = F.dropout(x, p=self.dropout, training=self.training)
  255.         x = self.classifier(x)
  256.         return x
  257.  
  258.  
  259. def ghostnet(**kwargs):
  260.     """
  261.     Constructs a GhostNet model
  262.     """
  263.     cfgs = [
  264.         # k, t, c, SE, s
  265.         # stage1
  266.         [[3, 16, 16, 0, 1]],
  267.         # stage2
  268.         [[3, 48, 24, 0, 2]],
  269.         [[3, 72, 24, 0, 1]],
  270.         # stage3
  271.         [[5, 72, 40, 0.25, 2]],
  272.         [[5, 120, 40, 0.25, 1]],
  273.         # stage4
  274.         [[3, 240, 80, 0, 2]],
  275.         [[3, 200, 80, 0, 1],
  276.          [3, 184, 80, 0, 1],
  277.          [3, 184, 80, 0, 1],
  278.          [3, 480, 112, 0.25, 1],
  279.          [3, 672, 112, 0.25, 1]
  280.          ],
  281.         # stage5
  282.         [[5, 672, 160, 0.25, 2]],
  283.         [[5, 960, 160, 0, 1],
  284.          [5, 960, 160, 0.25, 1],
  285.          [5, 960, 160, 0, 1],
  286.          [5, 960, 160, 0.25, 1]
  287.          ]
  288.     ]
  289.     return GhostNet(cfgs, **kwargs)
  290.

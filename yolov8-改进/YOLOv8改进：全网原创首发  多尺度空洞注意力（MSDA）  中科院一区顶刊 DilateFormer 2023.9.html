  26. class DilateAttention(nn.Module):
  27.     "Implementation of Dilate-attention"
  28.     def __init__(self, head_dim, qk_scale=None, attn_drop=0, kernel_size=3, dilation=1):
  29.         super().__init__()
  30.         self.head_dim = head_dim
  31.         self.scale = qk_scale or head_dim ** -0.5
  32.         self.kernel_size=kernel_size
  33.         self.unfold = nn.Unfold(kernel_size, dilation, dilation*(kernel_size-1)//2, 1)
  34.         self.attn_drop = nn.Dropout(attn_drop)
  35.  
  36.     def forward(self,q,k,v):
  37.         #B, C//3, H, W
  38.         B,d,H,W = q.shape
  39.  
  40.         q = q.reshape([B, d//self.head_dim, self.head_dim, 1 ,H*W]).permute(0, 1, 4, 3, 2)  # B,h,N,1,d
  41.         k = self.unfold(k).reshape([B, d//self.head_dim, self.head_dim, self.kernel_size*self.kernel_size, H*W]).permute(0, 1, 4, 2, 3)  #B,h,N,d,k*k
  42.         attn = (q @ k) * self.scale  # B,h,N,1,k*k
  43.         attn = attn.softmax(dim=-1)
  44.         attn = self.attn_drop(attn)
  45.         v = self.unfold(v).reshape([B, d//self.head_dim, self.head_dim, self.kernel_size*self.kernel_size, H*W]).permute(0, 1, 4, 3, 2)  # B,h,N,k*k,d
  46.         x = (attn @ v).transpose(1, 2).reshape(B, H, W, d)
  47.         return x

  165. class Bottleneck(nn.Module):
  166.     # Standard bottleneck
  167.     def __init__(self, c1, c2, shortcut=True, g=1, k=(3, 3), e=0.5):  # ch_in, ch_out, shortcut, groups, kernels, expand
  168.         super().__init__()

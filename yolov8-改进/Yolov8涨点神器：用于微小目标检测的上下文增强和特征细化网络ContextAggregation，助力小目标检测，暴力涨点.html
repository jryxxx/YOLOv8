  3. from mmcv.cnn import ConvModule
  4. from mmengine.model import caffe2_xavier_init, constant_init
  5. class ContextAggregation(nn.Module):
  6.     """
  7.     Context Aggregation Block.
  9.     Args:
  10.         in_channels (int): Number of input channels.
  11.         reduction (int, optional): Channel reduction ratio. Default: 1.
  12.         conv_cfg (dict or None, optional): Config dict for the convolution
  13.             layer. Default: None.
  14.     """
  15.  
  16.     def __init__(self, in_channels, reduction=1):
  17.         super(ContextAggregation, self).__init__()
  18.         self.in_channels = in_channels
  19.         self.reduction = reduction
  20.         self.inter_channels = max(in_channels // reduction, 1)
  21.  
  22.         conv_params = dict(kernel_size=1, act_cfg=None)
  23.  
  24.         self.a = ConvModule(in_channels, 1, **conv_params)
  25.         self.k = ConvModule(in_channels, 1, **conv_params)
  26.         self.v = ConvModule(in_channels, self.inter_channels, **conv_params)
  27.         self.m = ConvModule(self.inter_channels, in_channels, **conv_params)
  28.  
  29.         self.init_weights()
  30.  
  31.     def init_weights(self):
  32.         for m in (self.a, self.k, self.v):
  33.             caffe2_xavier_init(m.conv)
  34.         constant_init(self.m.conv, 0)
  35.  
  36.     def forward(self, x):
  37.         #n, c = x.size(0)
  38.         n = x.size(0)
  39.         c = self.inter_channels
  40.         #n, nH, nW, c = x.shape
  41.  
  42.         # a: [N, 1, H, W]
  43.         a = self.a(x).sigmoid()
  44.  
  45.         # k: [N, 1, HW, 1]
  46.         k = self.k(x).view(n, 1, -1, 1).softmax(2)
  47.  
  48.         # v: [N, 1, C, HW]
  49.         v = self.v(x).view(n, 1, c, -1)
  50.  
  51.         # y: [N, C, 1, 1]
  52.         y = torch.matmul(v, k).view(n, c, 1, 1)
  53.         y = self.m(y) * a
  54.  
  55.         return x + y

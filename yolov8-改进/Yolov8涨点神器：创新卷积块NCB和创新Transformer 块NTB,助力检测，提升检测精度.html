  14. class ConvBNReLU(nn.Module):
  15.     def __init__(
  16.             self,
  17.             in_channels,
  18.             out_channels,
  19.             kernel_size,
  20.             stride,
  21.             groups=1):
  22.         super(ConvBNReLU, self).__init__()
  23.         self.conv = nn.Conv2d(in_channels, out_channels, kernel_size=kernel_size, stride=stride,
  24.                               padding=1, groups=groups, bias=False)
  25.         self.norm = nn.BatchNorm2d(out_channels, eps=NORM_EPS)
  26.         self.act = nn.ReLU(inplace=True)
  27.  
  28.     def forward(self, x):

  30.         x = self.norm(x)
  31.         x = self.act(x)
  32.         return x

  35. def _make_divisible(v, divisor, min_value=None):
  36.     if min_value is None:
  37.         min_value = divisor
  38.     new_v = max(min_value, int(v + divisor / 2) // divisor * divisor)
  39.     # Make sure that round down does not go down by more than 10%.
  40.     if new_v < 0.9 * v:
  41.         new_v += divisor
  42.     return new_v

  45. class PatchEmbed(nn.Module):
  46.     def __init__(self,
  47.                  in_channels,
  48.                  out_channels,
  49.                  stride=1):
  50.         super(PatchEmbed, self).__init__()
  51.         norm_layer = partial(nn.BatchNorm2d, eps=NORM_EPS)
  52.         if stride == 2:
  53.             self.avgpool = nn.AvgPool2d((2, 2), stride=2, ceil_mode=True, count_include_pad=False)
  54.             self.conv = nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, bias=False)
  55.             self.norm = norm_layer(out_channels)
  56.         elif in_channels != out_channels:
  57.             self.avgpool = nn.Identity()
  58.             self.conv = nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, bias=False)
  59.             self.norm = norm_layer(out_channels)
  60.         else:
  61.             self.avgpool = nn.Identity()
  62.             self.conv = nn.Identity()
  63.             self.norm = nn.Identity()
  64.  
  65.     def forward(self, x):
  66.         return self.norm(self.conv(self.avgpool(x)))

  69. class MHCA(nn.Module):
  70.     """
  71.     Multi-Head Convolutional Attention
  72.     """
  73.     def __init__(self, out_channels, head_dim):
  74.         super(MHCA, self).__init__()
  75.         norm_layer = partial(nn.BatchNorm2d, eps=NORM_EPS)
  76.         self.group_conv3x3 = nn.Conv2d(out_channels, out_channels, kernel_size=3, stride=1,
  77.                                        padding=1, groups=out_channels // head_dim, bias=False)
  78.         self.norm = norm_layer(out_channels)
  79.         self.act = nn.ReLU(inplace=True)
  80.         self.projection = nn.Conv2d(out_channels, out_channels, kernel_size=1, bias=False)
  81.  
  82.     def forward(self, x):
  83.         out = self.group_conv3x3(x)
  84.         out = self.norm(out)
  85.         out = self.act(out)
  86.         out = self.projection(out)
  87.         return out

  90. class Mlp(nn.Module):
  91.     def __init__(self, in_features, out_features=None, mlp_ratio=None, drop=0., bias=True):
  92.         super().__init__()
  93.         out_features = out_features or in_features
  94.         hidden_dim = _make_divisible(in_features * mlp_ratio, 32)
  95.         self.conv1 = nn.Conv2d(in_features, hidden_dim, kernel_size=1, bias=bias)
  96.         self.act = nn.ReLU(inplace=True)
  97.         self.conv2 = nn.Conv2d(hidden_dim, out_features, kernel_size=1, bias=bias)
  98.         self.drop = nn.Dropout(drop)
  99.  
  100.     def forward(self, x):
  101.         x = self.conv1(x)
  102.         x = self.act(x)
  103.         x = self.drop(x)
  104.         x = self.conv2(x)
  105.         x = self.drop(x)
  106.         return x

  109. class NCB(nn.Module):
  110.     """
  111.     Next Convolution Block
  112.     """
  113.     def __init__(self, in_channels, out_channels, stride=1, path_dropout=0,
  114.                  drop=0, head_dim=32, mlp_ratio=3):
  115.         super(NCB, self).__init__()
  116.         self.in_channels = in_channels
  117.         self.out_channels = out_channels
  118.         norm_layer = partial(nn.BatchNorm2d, eps=NORM_EPS)
  119.         assert out_channels % head_dim == 0
  120.  
  121.         self.patch_embed = PatchEmbed(in_channels, out_channels, stride)
  122.         self.mhca = MHCA(out_channels, head_dim)
  123.         self.attention_path_dropout = DropPath(path_dropout)
  124.  
  125.         self.norm = norm_layer(out_channels)
  126.         self.mlp = Mlp(out_channels, mlp_ratio=mlp_ratio, drop=drop, bias=True)
  127.         self.mlp_path_dropout = DropPath(path_dropout)
  128.         self.is_bn_merged = False
  129.  
  130.     def forward(self, x):
  131.         x = self.patch_embed(x)
  132.         x = x + self.attention_path_dropout(self.mhca(x))
  133.         if not torch.onnx.is_in_onnx_export() and not self.is_bn_merged:
  134.             out = self.norm(x)
  135.         else:
  136.             out = x
  137.         x = x + self.mlp_path_dropout(self.mlp(out))
  138.         return x

  141. class E_MHSA(nn.Module):
  142.     """
  143.     Efficient Multi-Head Self Attention
  144.     """
  145.     def __init__(self, dim, out_dim=None, head_dim=32, qkv_bias=True, qk_scale=None,
  146.                  attn_drop=0, proj_drop=0., sr_ratio=1):
  147.         super().__init__()
  148.         self.dim = dim
  149.         self.out_dim = out_dim if out_dim is not None else dim
  150.         self.num_heads = self.dim // head_dim
  151.         self.scale = qk_scale or head_dim ** -0.5
  152.         self.q = nn.Linear(dim, self.dim, bias=qkv_bias)
  153.         self.k = nn.Linear(dim, self.dim, bias=qkv_bias)
  154.         self.v = nn.Linear(dim, self.dim, bias=qkv_bias)
  155.         self.proj = nn.Linear(self.dim, self.out_dim)
  156.         self.attn_drop = nn.Dropout(attn_drop)
  157.         self.proj_drop = nn.Dropout(proj_drop)
  158.  
  159.         self.sr_ratio = sr_ratio
  160.         self.N_ratio = sr_ratio ** 2
  161.         if sr_ratio > 1:
  162.             self.sr = nn.AvgPool1d(kernel_size=self.N_ratio, stride=self.N_ratio)
  163.             self.norm = nn.BatchNorm1d(dim, eps=NORM_EPS)
  164.         self.is_bn_merged = False
  165.  
  166.     def forward(self, x):
  167.         B, N, C = x.shape
  168.         q = self.q(x)
  169.         q = q.reshape(B, N, self.num_heads, int(C // self.num_heads)).permute(0, 2, 1, 3)
  170.  
  171.         if self.sr_ratio > 1:
  172.             x_ = x.transpose(1, 2)
  173.             x_ = self.sr(x_)
  174.             if not torch.onnx.is_in_onnx_export() and not self.is_bn_merged:
  175.                 x_ = self.norm(x_)
  176.             x_ = x_.transpose(1, 2)
  177.             k = self.k(x_)
  178.             k = k.reshape(B, -1, self.num_heads, int(C // self.num_heads)).permute(0, 2, 3, 1)
  179.             v = self.v(x_)
  180.             v = v.reshape(B, -1, self.num_heads, int(C // self.num_heads)).permute(0, 2, 1, 3)
  181.         else:
  182.             k = self.k(x)
  183.             k = k.reshape(B, -1, self.num_heads, int(C // self.num_heads)).permute(0, 2, 3, 1)
  184.             v = self.v(x)
  185.             v = v.reshape(B, -1, self.num_heads, int(C // self.num_heads)).permute(0, 2, 1, 3)
  186.         attn = (q @ k) * self.scale
  187.  
  188.         attn = attn.softmax(dim=-1)
  189.         attn = self.attn_drop(attn)
  190.  
  191.         x = (attn @ v).transpose(1, 2).reshape(B, N, C)
  192.         x = self.proj(x)
  193.         x = self.proj_drop(x)
  194.         return x

  197. class NTB(nn.Module):
  198.     """
  199.     Next Transformer Block
  200.     """
  201.     def __init__(
  202.             self, in_channels, out_channels, path_dropout=0.1, stride=1, sr_ratio=1,
  203.             mlp_ratio=2, head_dim=32, mix_block_ratio=0.75, attn_drop=0, drop=0,
  204.     ):
  205.         super(NTB, self).__init__()
  206.         self.in_channels = in_channels
  207.         self.out_channels = out_channels
  208.         self.mix_block_ratio = mix_block_ratio
  209.         norm_func = partial(nn.BatchNorm2d, eps=NORM_EPS)
  210.  
  211.         self.mhsa_out_channels = _make_divisible(int(out_channels * mix_block_ratio), 32)
  212.         self.mhca_out_channels = out_channels - self.mhsa_out_channels
  213.  
  214.         self.patch_embed = PatchEmbed(in_channels, self.mhsa_out_channels, stride)
  215.         self.norm1 = norm_func(self.mhsa_out_channels)
  216.         self.e_mhsa = E_MHSA(self.mhsa_out_channels, head_dim=head_dim, sr_ratio=sr_ratio,
  217.                              attn_drop=attn_drop, proj_drop=drop)
  218.         self.mhsa_path_dropout = DropPath(path_dropout * mix_block_ratio)
  219.  
  220.         self.projection = PatchEmbed(self.mhsa_out_channels, self.mhca_out_channels, stride=1)
  221.         self.mhca = MHCA(self.mhca_out_channels, head_dim=head_dim)
  222.         self.mhca_path_dropout = DropPath(path_dropout * (1 - mix_block_ratio))
  223.  
  224.         self.norm2 = norm_func(out_channels)
  225.         self.mlp = Mlp(out_channels, mlp_ratio=mlp_ratio, drop=drop)
  226.         self.mlp_path_dropout = DropPath(path_dropout)
  227.  
  228.         self.is_bn_merged = False
  229.  
  230.     def forward(self, x):
  231.         x = self.patch_embed(x)
  232.         B, C, H, W = x.shape
  233.         if not torch.onnx.is_in_onnx_export() and not self.is_bn_merged:
  234.             out = self.norm1(x)
  235.         else:
  236.             out = x
  237.         out = rearrange(out, "b c h w -> b (h w) c")  # b n c
  238.         out = self.mhsa_path_dropout(self.e_mhsa(out))
  239.         x = x + rearrange(out, "b (h w) c -> b c h w", h=H)
  240.  
  241.         out = self.projection(x)
  242.         out = out + self.mhca_path_dropout(self.mhca(out))
  243.         x = torch.cat([x, out], dim=1)
  244.  
  245.         if not torch.onnx.is_in_onnx_export() and not self.is_bn_merged:
  246.             out = self.norm2(x)
  247.         else:
  248.             out = x
  249.         x = x + self.mlp_path_dropout(self.mlp(out))
  250.         return x

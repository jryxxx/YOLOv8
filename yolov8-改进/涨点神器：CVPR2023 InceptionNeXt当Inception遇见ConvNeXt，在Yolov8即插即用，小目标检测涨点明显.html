1. InceptionNeXt介绍

摘要：受ViT的 long-range 建模能力的启发，大核卷积来扩大感受野用于提升模型性能，比如ConvNeXt了采用7x7深度卷积。虽然这种深度操作符只消耗少量FLOPs，但高内存访问成本，它在很大程度上损害了强大计算设备上的模型效率。为了解决这个问题，我们提出将大核深度卷积分解为沿通道维度的四个并行分支，即小平方内核、两个正交带核和一个恒等映射。通过这种新的Inception深度卷积，我们构建了一系列网络，即IncepitonNeXt，它们不仅具有高吞吐量，而且保持了有竞争力的性能。

       图 1：准确性和训练之间的权衡吞吐量。 所有模型均在 DeiT 训练下进行训练超参数 [61、37、38、69]。 训练吞吐量在批量大小为 128 的 A100 GPU 上测量。ConvNeXt-T/kn 表示具有深度卷积的变体内核大小为 n × n。 InceptionNeXt-T 兼具两者ResNet-50 的速度和 ConvNeXt-T 的准确性。

 

图 2：MetaFormer、MetaNext、ConvNeXt 和 InceptionNeXt 的框图。 

 1.1 Inception depthwise convolution

     将Inception的思想和ConvNeXt的设计相结合，从而实现了对大核深度卷积的有效分解。这种分解不仅减少了参数量和计算量，而且保留了大核深度卷积的优势，即扩大了感受野和提高了模型性能。

 Inception Depthwise Convolution 和其他几种算子的计算复杂度比较如下图3所示。可以看出，比普通卷积的效率要高得多。

  1. """
  2. InceptionNeXt implementation, paper: https://arxiv.org/abs/2303.16900
  4. Some code is borrowed from timm: https://github.com/huggingface/pytorch-image-models
  5. """
  6.  
  7. from functools import partial
  8.  
  9. import torch
  10. import torch.nn as nn
  11.  
  12. from timm.data import IMAGENET_DEFAULT_MEAN, IMAGENET_DEFAULT_STD
  13. from timm.models.helpers import checkpoint_seq
  14. from timm.models.layers import trunc_normal_, DropPath
  15. from timm.models.registry import register_model
  16. from timm.models.layers.helpers import to_2tuple
  17.  
  18.  
  19. class InceptionDWConv2d(nn.Module):
  20.     """ Inception depthweise convolution
  21.     """
  22.  
  23.     def __init__(self, in_channels, square_kernel_size=3, band_kernel_size=11, branch_ratio=0.125):
  24.         super().__init__()
  25.  
  26.         gc = int(in_channels * branch_ratio)  # channel numbers of a convolution branch
  27.         self.dwconv_hw = nn.Conv2d(gc, gc, square_kernel_size, padding=square_kernel_size // 2, groups=gc)
  28.         self.dwconv_w = nn.Conv2d(gc, gc, kernel_size=(1, band_kernel_size), padding=(0, band_kernel_size // 2),
  29.                                   groups=gc)
  30.         self.dwconv_h = nn.Conv2d(gc, gc, kernel_size=(band_kernel_size, 1), padding=(band_kernel_size // 2, 0),
  31.                                   groups=gc)
  32.         self.split_indexes = (in_channels - 3 * gc, gc, gc, gc)
  33.  
  34.     def forward(self, x):
  35.         x_id, x_hw, x_w, x_h = torch.split(x, self.split_indexes, dim=1)
  36.         return torch.cat(
  37.             (x_id, self.dwconv_hw(x_hw), self.dwconv_w(x_w), self.dwconv_h(x_h)),
  38.             dim=1,
  39.         )
  40.  
  41.  
  42. class ConvMlp(nn.Module):
  43.     """ MLP using 1x1 convs that keeps spatial dims
  44.     copied from timm: https://github.com/huggingface/pytorch-image-models/blob/v0.6.11/timm/models/layers/mlp.py
  45.     """
  46.  
  47.     def __init__(
  48.             self, in_features, hidden_features=None, out_features=None, act_layer=nn.ReLU,
  49.             norm_layer=None, bias=True, drop=0.):
  50.         super().__init__()
  51.         out_features = out_features or in_features
  52.         hidden_features = hidden_features or in_features
  53.         bias = to_2tuple(bias)
  54.  
  55.         self.fc1 = nn.Conv2d(in_features, hidden_features, kernel_size=1, bias=bias[0])
  56.         self.norm = norm_layer(hidden_features) if norm_layer else nn.Identity()
  57.         self.act = act_layer()
  58.         self.drop = nn.Dropout(drop)
  59.         self.fc2 = nn.Conv2d(hidden_features, out_features, kernel_size=1, bias=bias[1])
  60.  
  61.     def forward(self, x):
  62.         x = self.fc1(x)
  63.         x = self.norm(x)
  64.         x = self.act(x)
  65.         x = self.drop(x)
  66.         x = self.fc2(x)
  67.         return x
  68.  
  69.  
  70. class MlpHead(nn.Module):
  71.     """ MLP classification head
  72.     """
  73.  
  74.     def __init__(self, dim, num_classes=1000, mlp_ratio=3, act_layer=nn.GELU,
  75.                  norm_layer=partial(nn.LayerNorm, eps=1e-6), drop=0., bias=True):
  76.         super().__init__()
  77.         hidden_features = int(mlp_ratio * dim)
  78.         self.fc1 = nn.Linear(dim, hidden_features, bias=bias)
  79.         self.act = act_layer()
  80.         self.norm = norm_layer(hidden_features)
  81.         self.fc2 = nn.Linear(hidden_features, num_classes, bias=bias)
  82.         self.drop = nn.Dropout(drop)
  83.  
  84.     def forward(self, x):
  85.         x = x.mean((2, 3))  # global average pooling
  86.         x = self.fc1(x)
  87.         x = self.act(x)
  88.         x = self.norm(x)
  89.         x = self.drop(x)
  90.         x = self.fc2(x)
  91.         return x
  92.  
  93.  
  94. class MetaNeXtBlock(nn.Module):
  95.     """ MetaNeXtBlock Block
  96.     Args:
  97.         dim (int): Number of input channels.
  98.         drop_path (float): Stochastic depth rate. Default: 0.0
  99.         ls_init_value (float): Init value for Layer Scale. Default: 1e-6.
  100.     """
  101.  
  102.     def __init__(
  103.             self,
  104.             dim,
  105.             token_mixer=InceptionDWConv2d,
  106.             norm_layer=nn.BatchNorm2d,
  107.             mlp_layer=ConvMlp,
  108.             mlp_ratio=4,
  109.             act_layer=nn.GELU,
  110.             ls_init_value=1e-6,
  111.             drop_path=0.,
  112.  
  113.     ):
  114.         super().__init__()
  115.         self.token_mixer = token_mixer(dim)
  116.         self.norm = norm_layer(dim)
  117.         self.mlp = mlp_layer(dim, int(mlp_ratio * dim), act_layer=act_layer)
  118.         self.gamma = nn.Parameter(ls_init_value * torch.ones(dim)) if ls_init_value else None
  119.         self.drop_path = DropPath(drop_path) if drop_path > 0. else nn.Identity()
  120.  
  121.     def forward(self, x):
  122.         shortcut = x
  123.         x = self.token_mixer(x)
  124.         x = self.norm(x)
  125.         x = self.mlp(x)
  126.         if self.gamma is not None:
  127.             x = x.mul(self.gamma.reshape(1, -1, 1, 1))
  128.         x = self.drop_path(x) + shortcut
  129.         return x
  130.  
  131. class MetaNeXtStage(nn.Module):
  132.     def __init__(
  133.             self,
  134.             in_chs,
  135.             out_chs,
  136.             ds_stride=2,
  137.             depth=1,
  138.             drop_path_rates=None,
  139.             ls_init_value=1.0,
  140.             act_layer=nn.GELU,
  141.             norm_layer=nn.BatchNorm2d,
  142.             mlp_ratio=4,
  143.     ):
  144.         super().__init__()
  145.         self.grad_checkpointing = False
  146.         if ds_stride > 1:
  147.             self.downsample = nn.Sequential(
  148.                 norm_layer(in_chs),
  149.                 nn.Conv2d(in_chs, out_chs, kernel_size=ds_stride, stride=ds_stride),
  150.             )
  151.         else:
  152.             self.downsample = nn.Identity()
  153.  
  154.         drop_path_rates = drop_path_rates or [0.] * depth
  155.         stage_blocks = []
  156.         for i in range(depth):
  157.             stage_blocks.append(MetaNeXtBlock(
  158.                 dim=out_chs,
  159.                 drop_path=drop_path_rates[i],
  160.                 ls_init_value=ls_init_value,
  161.                 act_layer=act_layer,
  162.                 norm_layer=norm_layer,
  163.                 mlp_ratio=mlp_ratio,
  164.             ))
  165.             in_chs = out_chs
  166.         self.blocks = nn.Sequential(*stage_blocks)
  167.  
  168.     def forward(self, x):
  169.         x = self.downsample(x)
  170.         if self.grad_checkpointing and not torch.jit.is_scripting():
  171.             x = checkpoint_seq(self.blocks, x)
  172.         else:
  173.             x = self.blocks(x)
  174.         return x

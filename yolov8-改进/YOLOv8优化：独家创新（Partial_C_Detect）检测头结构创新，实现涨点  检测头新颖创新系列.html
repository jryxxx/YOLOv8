1.Partial_C_Detect原理介绍

为了设计快速神经网络，许多工作都集中在减少浮点运算（FLOPs）的数量上。然而，作者观察到FLOPs的这种减少不一定会带来延迟的类似程度的减少。这主要源于每秒低浮点运算（FLOPS）效率低下。为了实现更快的网络，作者重新回顾了FLOPs的运算符，并证明了如此低的FLOPS主要是由于运算符的频繁内存访问，尤其是深度卷积。因此，本文提出了一种新的partial convolution（PConv），通过同时减少冗余计算和内存访问可以更有效地提取空间特征。

      基于PConv进一步提出FasterNet，这是一个新的神经网络家族，它在广泛的设备上实现了比其他网络高得多的运行速度，而不影响各种视觉任务的准确性。例如，在ImageNet-1k上小型FasterNet-T0在GPU、CPU和ARM处理器上分别比MobileVitXXS快3.1倍、3.1倍和2.5倍，同时准确度提高2.9%。
     又快又好！本文提出新的Partial卷积(PConv)，同时减少冗余计算和内存访问，并进一步提出FasterNet：新的神经网络家族，在多个处理平台上运行速度更快，优于MobileVit等网络；

  24. class Partial_conv3(nn.Module):
  25.     def __init__(self, dim,  n_div=4, forward='split_cat'):
  26.         super().__init__()
  27.         self.dim_conv3 = dim // n_div
  28.         self.dim_untouched = dim - self.dim_conv3
  29.         self.partial_conv3 = nn.Conv2d(self.dim_conv3, self.dim_conv3, 3, 1, 1, bias=False)
  30.  
  31.         if forward == 'slicing':
  32.             self.forward = self.forward_slicing
  33.         elif forward == 'split_cat':
  34.             self.forward = self.forward_split_cat
  35.         else:
  36.             raise NotImplementedError
  37.  
  38.     def forward_slicing(self, x):
  39.         # only for inference
  40.         x = x.clone()   # !!! Keep the original input intact for the residual connection later
  41.         x[:, :self.dim_conv3, :, :] = self.partial_conv3(x[:, :self.dim_conv3, :, :])
  42.         return x
  43.  
  44.     def forward_split_cat(self, x):
  45.         # for training/inference
  46.         x1, x2 = torch.split(x, [self.dim_conv3, self.dim_untouched], dim=1)
  47.         x1 = self.partial_conv3(x1)
  48.         x = torch.cat((x1, x2), 1)
  49.         return x

  52. class Detect_improve(nn.Module):
  53.     """YOLOv8 Detect Efficient head for detection models."""
  54.     dynamic = False  # force grid reconstruction
  55.     export = False  # export mode
  56.     shape = None
  57.     anchors = torch.empty(0)  # init
  58.     strides = torch.empty(0)  # init
  59.  
  60.     def __init__(self, nc=80, ch=()):  # detection layer
  61.         super().__init__()
  62.         self.nc = nc  # number of classes
  63.         self.nl = len(ch)  # number of detection layers
  64.         self.reg_max = 16  # DFL channels (ch[0] // 16 to scale 4/8/12/16/20 for n/s/m/l/x)
  65.         self.no = nc + self.reg_max * 4  # number of outputs per anchor
  66.         self.stride = torch.zeros(self.nl)  # strides computed during build
  67.  
  68.         self.stem = nn.ModuleList(nn.Sequential(Partial_conv3(x, 4), Conv(x, x, 1)) for x in ch) 
  69.         self.cv2 = nn.ModuleList(nn.Conv2d(x, 4 * self.reg_max, 1) for x in ch)
  70.         self.cv3 = nn.ModuleList(nn.Conv2d(x, self.nc, 1) for x in ch)
  71.         self.dfl = DFL(self.reg_max) if self.reg_max > 1 else nn.Identity()
  72.  
  73.     def forward(self, x):
  74.         """Concatenates and returns predicted bounding boxes and class probabilities."""
  75.         shape = x[0].shape  # BCHW
  76.         for i in range(self.nl):
  77.             x[i] = self.stem[i](x[i])
  78.             x[i] = torch.cat((self.cv2[i](x[i]), self.cv3[i](x[i])), 1)
  79.         if self.training:
  80.             return x
  81.         elif self.dynamic or self.shape != shape:
  82.             self.anchors, self.strides = (x.transpose(0, 1) for x in make_anchors(x, self.stride, 0.5))
  83.             self.shape = shape
  84.  
  85.         x_cat = torch.cat([xi.view(shape[0], self.no, -1) for xi in x], 2)
  86.         if self.export and self.format in ('saved_model', 'pb', 'tflite', 'edgetpu', 'tfjs'):  # avoid TF FlexSplitV ops
  87.             box = x_cat[:, :self.reg_max * 4]
  88.             cls = x_cat[:, self.reg_max * 4:]
  89.         else:
  90.             box, cls = x_cat.split((self.reg_max * 4, self.nc), 1)
  91.         dbox = dist2bbox(self.dfl(box), self.anchors.unsqueeze(0), xywh=True, dim=1) * self.strides
  92.         y = torch.cat((dbox, cls.sigmoid()), 1)
  93.         return y if self.export else (y, x)
  94.

  96.         """Initialize Detect() biases, WARNING: requires stride availability."""

  101.             a.bias.data[:] = 1.0  # box

  9. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  10.     # Pad to 'same' shape outputs
  11.     if d > 1:
  12.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  13.     if p is None:
  14.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  15.     return p

  18. class Conv(nn.Module):
  19.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  20.     default_act = nn.SiLU()  # default activation
  21.  
  22.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  23.         super().__init__()
  24.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  25.         self.bn = nn.BatchNorm2d(c2)
  26.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  27.  
  28.     def forward(self, x):
  29.         return self.act(self.bn(self.conv(x)))
  30.  
  31.     def forward_fuse(self, x):
  32.         return self.act(self.conv(x))

  70. class MixConv2d(nn.Module):
  71.     # Mixed Depth-wise Conv https://arxiv.org/abs/1907.09595
  72.     def __init__(self, c1, c2, k=(1, 3), s=1, equal_ch=True):  # ch_in, ch_out, kernel, stride, ch_strategy
  73.         super().__init__()
  74.         n = len(k)  # number of convolutions
  75.         if equal_ch:  # equal c_ per group
  76.             i = torch.linspace(0, n - 1E-6, c2).floor()  # c2 indices
  77.             c_ = [(i == g).sum() for g in range(n)]  # intermediate channels
  78.         else:  # equal weight.numel() per group
  79.             b = [c2] + [0] * n
  80.             a = np.eye(n + 1, n, k=-1)
  81.             a -= np.roll(a, 1, axis=1)
  82.             a *= np.array(k) ** 2
  83.             a[0] = 1
  84.             c_ = np.linalg.lstsq(a, b, rcond=None)[0].round()  # solve for equal weight indices, ax = b
  85.  
  86.         self.m = nn.ModuleList([
  87.             nn.Conv2d(c1, int(c_), k, s, k // 2, groups=math.gcd(c1, int(c_)), bias=False) for k, c_ in zip(k, c_)])
  88.         self.bn = nn.BatchNorm2d(c2)
  89.         self.act = nn.SiLU()
  90.  
  91.     def forward(self, x):
  92.         return self.act(self.bn(torch.cat([m(x) for m in self.m], 1)))

  95. class CAConv(nn.Module):
  96.     def __init__(self, inp, oup, kernel_size, stride, reduction=32):
  97.         super(CAConv, self).__init__()
  98.         self.pool_h = nn.AdaptiveAvgPool2d((None, 1))
  99.         self.pool_w = nn.AdaptiveAvgPool2d((1, None))
  100.  
  101.         mip = max(8, inp // reduction)
  102.  
  103.         self.conv1 = nn.Conv2d(inp, mip, kernel_size=1, stride=1, padding=0)
  104.         self.bn1 = nn.BatchNorm2d(mip)
  105.         self.act = h_swish()
  106.  
  107.         self.conv_h = nn.Conv2d(mip, inp, kernel_size=1, stride=1, padding=0)
  108.         self.conv_w = nn.Conv2d(mip, inp, kernel_size=1, stride=1, padding=0)
  109.         self.conv = nn.Sequential(nn.Conv2d(inp, oup, kernel_size, padding=kernel_size // 2, stride=stride),
  110.                                   nn.BatchNorm2d(oup),
  111.                                   nn.ReLU())
  112.  
  113.     def forward(self, x):
  114.         identity = x
  115.  
  116.         n, c, h, w = x.size()
  117.         x_h = self.pool_h(x)
  118.         x_w = self.pool_w(x).permute(0, 1, 3, 2)
  119.  
  120.         y = torch.cat([x_h, x_w], dim=2)
  121.         y = self.conv1(y)
  122.         y = self.bn1(y)
  123.         y = self.act(y)
  124.  
  125.         x_h, x_w = torch.split(y, [h, w], dim=2)
  126.         x_w = x_w.permute(0, 1, 3, 2)
  127.  
  128.         a_h = self.conv_h(x_h).sigmoid()
  129.         a_w = self.conv_w(x_w).sigmoid()
  130.  
  131.         out = identity * a_w * a_h
  132.  
  133.         return self.conv(out)
  134. class CBAMConv(nn.Module):
  135.     def __init__(self, channel, out_channel, kernel_size, stride, reduction=16, spatial_kernel=7):
  136.         super().__init__()
  137.  
  138.         self.max_pool = nn.AdaptiveMaxPool2d(1)
  139.         self.avg_pool = nn.AdaptiveAvgPool2d(1)
  140.         self.mlp = nn.Sequential(
  141.             nn.Conv2d(channel, channel // reduction, 1, bias=False),
  142.             nn.ReLU(inplace=True),
  143.             nn.Conv2d(channel // reduction, channel, 1, bias=False)
  144.         )
  145.  
  146.         self.spatital = nn.Conv2d(2, 1, kernel_size=spatial_kernel,
  147.                                   padding=spatial_kernel // 2, bias=False)
  148.         self.sigmoid = nn.Sigmoid()
  149.  
  150.         self.conv = nn.Sequential(nn.Conv2d(channel, out_channel, kernel_size, padding=kernel_size // 2, stride=stride),
  151.                                   nn.BatchNorm2d(out_channel),
  152.                                   nn.ReLU())
  153.         kernel_sizes = [3, 5, 7]
  154.         self.attention_layers = nn.ModuleList([
  155.             nn.Sequential(
  156.                 nn.Conv2d(2, 1, kernel_size=k, padding=k // 2),
  157.                 nn.Sigmoid()
  158.             ) for k in kernel_sizes
  159.         ])
  160.  
  161.     def forward(self, x):
  162.         max_out = self.mlp(self.max_pool(x))
  163.         avg_out = self.mlp(self.avg_pool(x))
  164.         channel_out = self.sigmoid(max_out + avg_out)
  165.         x = channel_out * x
  166.  
  167.         max_out, _ = torch.max(x, dim=1, keepdim=True)
  168.         avg_out = torch.mean(x, dim=1, keepdim=True)
  169.         spatial_out = self.sigmoid(self.spatital(torch.cat([max_out, avg_out], dim=1)))
  170.         spatial_out = [att_layer(torch.cat([max_out, avg_out], dim=1)) for att_layer in self.attention_layers]
  171.         spatial_out = self.sigmoid(sum(spatial_out)) #/ len(spatial_out)
  172.         x = spatial_out * x
  173.         return self.conv(x)

  199. class RFCAConv(nn.Module):

  201.         super(RFCAConv, self).__init__()

  204.         self.group_conv2 = Conv(c1, 9 *c1, k=3, g=c1)

  215.  
  216.         group1 = self.softmax(self.group_conv1(y))
  217.         group2 = self.softmax(self.group_conv2(y))
  218.         group3 = self.softmax(self.group_conv3(y))

  221.         g2 = self.group_conv(x)
  222.  
  223.         out1 = g2 * group1
  224.         out2 = g2 * group2
  225.         out3 = g2 * group3
  226.

  228.         out = torch.cat([out1, out2, out3], dim=1)

  236.         # out = out.view(batch_size, output_channels, 3, 3, height, width).permute(0, 1, 4, 2, 5,3).\
  237.         #                                         reshape(batch_size, output_channels, 3 * height, 3 * width)
  238.         out = out.view(batch_size, output_channels, height*3, width*3)
  239.         out = self.convDown(out)

  242. class RFAConv(nn.Module):
  243.     def __init__(self, in_channels, out_channels, kernel_size, stride=1):
  244.         super(RFAConv, self).__init__()
  245.  
  246.         # Adaptive Average Pooling
  247.         self.avg_pool = nn.AdaptiveAvgPool2d(1)

  249.         # Non-shared Convolution layers

  251.         self.conv2 = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=kernel_size//2, groups=in_channels)
  252.         self.conv3 = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=kernel_size//2, groups=in_channels)

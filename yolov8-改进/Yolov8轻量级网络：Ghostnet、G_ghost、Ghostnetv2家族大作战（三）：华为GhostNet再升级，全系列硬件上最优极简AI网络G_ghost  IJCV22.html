  3. """
  4. Creates a G-Ghost-RegNet Model as defined in:
  5. GhostNets on Heterogeneous Devices via Cheap Operations
  6. By Kai Han, Yunhe Wang, Chang Xu, Jianyuan Guo, Chunjing Xu, Enhua Wu, Qi Tian.
  7. arXiv link: https://arxiv.org/abs/2201.03297
  8. Modified from https://github.com/d-li14/regnet.pytorch
  9. """
  10. import numpy as np
  11. import torch
  12. import torch.nn as nn
  13.  
  14. __all__ = ['g_ghost_regnetx_002', 'g_ghost_regnetx_004', 'g_ghost_regnetx_006', 'g_ghost_regnetx_008', 'g_ghost_regnetx_016', 'g_ghost_regnetx_032',
  15.            'g_ghost_regnetx_040', 'g_ghost_regnetx_064', 'g_ghost_regnetx_080', 'g_ghost_regnetx_120', 'g_ghost_regnetx_160', 'g_ghost_regnetx_320']
  16.  
  17.  
  18.  
  19. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  20.     # Pad to 'same' shape outputs
  21.     if d > 1:
  22.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  23.     if p is None:
  24.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  25.     return p
  26.  
  27.  
  28. class Conv(nn.Module):
  29.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  30.     default_act = nn.SiLU()  # default activation
  31.  
  32.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  33.         super().__init__()
  34.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  35.         self.bn = nn.BatchNorm2d(c2)
  36.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  37.  
  38.     def forward(self, x):
  39.         return self.act(self.bn(self.conv(x)))
  40.  
  41.     def forward_fuse(self, x):
  42.         return self.act(self.conv(x))
  43.  
  44. def conv3x3(in_planes, out_planes, stride=1, groups=1, dilation=1):
  45.     """3x3 convolution with padding"""
  46.     return nn.Conv2d(in_planes, out_planes, kernel_size=3, stride=stride,
  47.                      padding=dilation, groups=groups, bias=False, dilation=dilation)
  48.  
  49.  
  50. def conv1x1(in_planes, out_planes, stride=1):
  51.     """1x1 convolution"""
  52.     return nn.Conv2d(in_planes, out_planes, kernel_size=1, stride=stride, bias=False)
  53.  
  54.  
  55. class Bottleneck(nn.Module):
  56.     expansion = 1
  57.     __constants__ = ['downsample']
  58.  
  59.     def __init__(self, inplanes, planes, stride=1, downsample=None, group_width=1,
  60.                  dilation=1, norm_layer=None):
  61.         super(Bottleneck, self).__init__()
  62.         if norm_layer is None:
  63.             norm_layer = nn.BatchNorm2d
  64.         width = planes * self.expansion
  65.         # Both self.conv2 and self.downsample layers downsample the input when stride != 1
  66.         self.conv1 = conv1x1(inplanes, width)
  67.         self.bn1 = norm_layer(width)
  68.         self.conv2 = conv3x3(width, width, stride, width // min(width, group_width), dilation)
  69.         self.bn2 = norm_layer(width)
  70.         self.conv3 = conv1x1(width, planes)
  71.         self.bn3 = norm_layer(planes)
  72.         self.relu = nn.ReLU(inplace=True)
  73.         self.downsample = downsample
  74.         self.stride = stride
  75.  
  76.     def forward(self, x):
  77.         identity = x
  78.  
  79.         out = self.conv1(x)
  80.         out = self.bn1(out)
  81.         out = self.relu(out)
  82.  
  83.         out = self.conv2(out)
  84.         out = self.bn2(out)
  85.         out = self.relu(out)
  86.  
  87.         out = self.conv3(out)
  88.         out = self.bn3(out)
  89.  
  90.         if self.downsample is not None:
  91.             identity = self.downsample(x)
  92.  
  93.         out += identity
  94.         out = self.relu(out)
  95.  
  96.         return out
  97.  
  98.  
  99. class C2f_g_ghostBottleneck(nn.Module):
  100.     # CSP Bottleneck with 2 convolutions
  101.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  102.         super().__init__()
  103.         self.c = int(c2 * e)  # hidden channels
  104.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  105.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  106.         self.m = nn.ModuleList(Bottleneck(self.c, self.c) for _ in range(n))
  107.  
  108.     def forward(self, x):
  109.         y = list(self.cv1(x).chunk(2, 1))
  110.         y.extend(m(y[-1]) for m in self.m)
  111.         return self.cv2(torch.cat(y, 1))
  112.  
  113. class LambdaLayer(nn.Module):
  114.     def __init__(self, lambd):
  115.         super(LambdaLayer, self).__init__()
  116.         self.lambd = lambd
  117.  
  118.     def forward(self, x):
  119.         return self.lambd(x)
  120.

  122. class Stage(nn.Module):
  123.  
  124.     def __init__(self, block, inplanes, planes, group_width, blocks, stride=1, dilate=False, cheap_ratio=0.5):
  125.         super(Stage, self).__init__()
  126.         norm_layer = nn.BatchNorm2d
  127.         downsample = None
  128.         self.dilation = 1
  129.         previous_dilation = self.dilation
  130.         if dilate:
  131.             self.dilation *= stride
  132.             stride = 1
  133.         if stride != 1 or self.inplanes != planes:
  134.             downsample = nn.Sequential(
  135.                 conv1x1(inplanes, planes, stride),
  136.                 norm_layer(planes),
  137.             )
  138.             
  139.         self.base = block(inplanes, planes, stride, downsample, group_width,
  140.                             previous_dilation, norm_layer)
  141.         self.end = block(planes, planes, group_width=group_width,
  142.                                 dilation=self.dilation,
  143.                                 norm_layer=norm_layer)
  144.         
  145.         group_width = int(group_width * 0.75)
  146.         raw_planes = int(planes * (1 - cheap_ratio) / group_width) * group_width
  147.         cheap_planes = planes - raw_planes
  148.         self.cheap_planes = cheap_planes
  149.         self.raw_planes = raw_planes
  150.         
  151.         self.merge = nn.Sequential(
  152.             nn.AdaptiveAvgPool2d(1),
  153.             nn.Conv2d(planes+raw_planes*(blocks-2), cheap_planes,
  154.                       kernel_size=1, stride=1, bias=False),
  155.             nn.BatchNorm2d(cheap_planes),
  156.             nn.ReLU(inplace=True),
  157.             nn.Conv2d(cheap_planes, cheap_planes, kernel_size=1, bias=False),
  158.             nn.BatchNorm2d(cheap_planes),
  159.         )
  160.         self.cheap = nn.Sequential(
  161.             nn.Conv2d(cheap_planes, cheap_planes,
  162.                       kernel_size=1, stride=1, bias=False),
  163.             nn.BatchNorm2d(cheap_planes),
  164.         )
  165.         self.cheap_relu = nn.ReLU(inplace=True)
  166.         
  167.         layers = []
  168.         downsample = nn.Sequential(
  169.             LambdaLayer(lambda x: x[:, :raw_planes])
  170.         )
  171.  
  172.         layers = []
  173.         layers.append(block(raw_planes, raw_planes, 1, downsample, group_width,
  174.                             self.dilation, norm_layer))
  175.         inplanes = raw_planes
  176.         for _ in range(2, blocks-1):
  177.             layers.append(block(inplanes, raw_planes, group_width=group_width,
  178.                                 dilation=self.dilation,
  179.                                 norm_layer=norm_layer))
  180.  
  181.         self.layers = nn.Sequential(*layers)
  182.         
  183.     
  184.     def forward(self, input):
  185.         x0 = self.base(input)
  186.         
  187.         m_list = [x0]
  188.         e = x0[:, :self.raw_planes]        
  189.         for l in self.layers:
  190.             e = l(e)
  191.             m_list.append(e)
  192.         m = torch.cat(m_list,1)
  193.         m = self.merge(m)
  194.         
  195.         c = x0[:, self.raw_planes:]
  196.         c = self.cheap_relu(self.cheap(c)+m)
  197.         
  198.         x = torch.cat((e,c),1)
  199.         x = self.end(x)
  200.         return x
  201.

  203. class GGhostRegNet(nn.Module):
  204.  
  205.     def __init__(self, block, layers, widths, num_classes=1000, zero_init_residual=True,
  206.                  group_width=1, replace_stride_with_dilation=None,
  207.                  norm_layer=None):
  208.         super(GGhostRegNet, self).__init__()
  209.         if norm_layer is None:
  210.             norm_layer = nn.BatchNorm2d
  211.         self._norm_layer = norm_layer
  212.  
  213.         self.inplanes = 32
  214.         self.dilation = 1
  215.         if replace_stride_with_dilation is None:
  216.             # each element in the tuple indicates if we should replace
  217.             # the 2x2 stride with a dilated convolution instead
  218.             replace_stride_with_dilation = [False, False, False, False]
  219.         if len(replace_stride_with_dilation) != 4:
  220.             raise ValueError("replace_stride_with_dilation should be None "
  221.                              "or a 4-element tuple, got {}".format(replace_stride_with_dilation))
  222.         self.group_width = group_width
  223.         self.conv1 = nn.Conv2d(3, self.inplanes, kernel_size=3, stride=2, padding=1,
  224.                                bias=False)
  225.         self.bn1 = norm_layer(self.inplanes)
  226.         self.relu = nn.ReLU(inplace=True)
  227.         
  228.         self.layer1 = self._make_layer(block, widths[0], layers[0], stride=2,
  229.                                        dilate=replace_stride_with_dilation[0])
  230.         
  231.         self.inplanes = widths[0]
  232.         if layers[1] > 2:
  233.             self.layer2 = Stage(block, self.inplanes, widths[1], group_width, layers[1], stride=2,
  234.                           dilate=replace_stride_with_dilation[1], cheap_ratio=0.5) 
  235.         else:      
  236.             self.layer2 = self._make_layer(block, widths[1], layers[1], stride=2,
  237.                                            dilate=replace_stride_with_dilation[1])
  238.         
  239.         self.inplanes = widths[1]
  240.         self.layer3 = Stage(block, self.inplanes, widths[2], group_width, layers[2], stride=2,
  241.                       dilate=replace_stride_with_dilation[2], cheap_ratio=0.5)
  242.         
  243.         self.inplanes = widths[2]
  244.         if layers[3] > 2:
  245.             self.layer4 = Stage(block, self.inplanes, widths[3], group_width, layers[3], stride=2,
  246.                           dilate=replace_stride_with_dilation[3], cheap_ratio=0.5) 
  247.         else:
  248.             self.layer4 = self._make_layer(block, widths[3], layers[3], stride=2,
  249.                                            dilate=replace_stride_with_dilation[3])
  250.         self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
  251.         self.dropout = nn.Dropout(0.2)
  252.         self.fc = nn.Linear(widths[-1] * block.expansion, num_classes)
  253.  
  254.         for m in self.modules():
  255.             if isinstance(m, nn.Conv2d):
  256.                 nn.init.kaiming_normal_(m.weight, mode='fan_out', nonlinearity='relu')
  257.             elif isinstance(m, (nn.BatchNorm2d, nn.GroupNorm)):
  258.                 nn.init.constant_(m.weight, 1)
  259.                 nn.init.constant_(m.bias, 0)
  260.  
  261.     def _make_layer(self, block, planes, blocks, stride=1, dilate=False):
  262.         norm_layer = self._norm_layer
  263.         downsample = None
  264.         previous_dilation = self.dilation
  265.         if dilate:
  266.             self.dilation *= stride
  267.             stride = 1
  268.         if stride != 1 or self.inplanes != planes:
  269.             downsample = nn.Sequential(
  270.                 conv1x1(self.inplanes, planes, stride),
  271.                 norm_layer(planes),
  272.             )
  273.  
  274.         layers = []
  275.         layers.append(block(self.inplanes, planes, stride, downsample, self.group_width,
  276.                             previous_dilation, norm_layer))
  277.         self.inplanes = planes
  278.         for _ in range(1, blocks):
  279.             layers.append(block(self.inplanes, planes, group_width=self.group_width,
  280.                                 dilation=self.dilation,
  281.                                 norm_layer=norm_layer))
  282.  
  283.         return nn.Sequential(*layers)
  284.  
  285.     def _forward_impl(self, x):
  286.         # See note [TorchScript super()]
  287.         x = self.conv1(x)
  288.         x = self.bn1(x)
  289.         x = self.relu(x)
  290.  
  291.         x = self.layer1(x)
  292.         x = self.layer2(x)
  293.         x = self.layer3(x)
  294.         x = self.layer4(x)
  295.  
  296.         x = self.avgpool(x)
  297.         x = torch.flatten(x, 1)
  298.         x = self.dropout(x)
  299.         x = self.fc(x)
  300.  
  301.         return x

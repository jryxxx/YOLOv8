  2. class TridentBlock(nn.Module):
  3.     def __init__(self, c1, c2, stride=1, c=False, e=0.5, padding=[1, 2, 3], dilate=[1, 2, 3], bias=False):
  4.         super(TridentBlock, self).__init__()
  5.         self.stride = stride
  6.         self.c = c
  7.         c_ = int(c2 * e)
  8.         self.padding = padding
  9.         self.dilate = dilate
  10.         self.share_weightconv1 = nn.Parameter(torch.Tensor(c_, c1, 1, 1))
  11.         self.share_weightconv2 = nn.Parameter(torch.Tensor(c2, c_, 3, 3))
  12.  
  13.         self.bn1 = nn.BatchNorm2d(c_)
  14.         self.bn2 = nn.BatchNorm2d(c2)
  15.  
  16.         self.act = nn.SiLU()
  17.  
  18.         nn.init.kaiming_uniform_(self.share_weightconv1, nonlinearity="relu")
  19.         nn.init.kaiming_uniform_(self.share_weightconv2, nonlinearity="relu")
  20.  
  21.         if bias:
  22.             self.bias = nn.Parameter(torch.Tensor(c2))
  23.         else:
  24.             self.bias = None
  25.  
  26.         if self.bias is not None:
  27.             nn.init.constant_(self.bias, 0)
  28.  
  29.     def forward_for_small(self, x):
  30.         residual = x
  31.         out = nn.functional.conv2d(x, self.share_weightconv1, bias=self.bias)
  32.         out = self.bn1(out)
  33.         out = self.act(out)
  34.  
  35.         out = nn.functional.conv2d(out, self.share_weightconv2, bias=self.bias, stride=self.stride, padding=self.padding[0],
  36.                                    dilation=self.dilate[0])
  37.         out = self.bn2(out)
  38.         out += residual
  39.         out = self.act(out)
  40.  
  41.         return out
  42.  
  43.     def forward_for_middle(self, x):
  44.         residual = x
  45.         out = nn.functional.conv2d(x, self.share_weightconv1, bias=self.bias)
  46.         out = self.bn1(out)
  47.         out = self.act(out)
  48.  
  49.         out = nn.functional.conv2d(out, self.share_weightconv2, bias=self.bias, stride=self.stride, padding=self.padding[1],
  50.                                    dilation=self.dilate[1])
  51.         out = self.bn2(out)
  52.         out += residual
  53.         out = self.act(out)
  54.  
  55.         return out
  56.  
  57.     def forward_for_big(self, x):
  58.         residual = x
  59.         out = nn.functional.conv2d(x, self.share_weightconv1, bias=self.bias)
  60.         out = self.bn1(out)
  61.         out = self.act(out)
  62.  
  63.         out = nn.functional.conv2d(out, self.share_weightconv2, bias=self.bias, stride=self.stride, padding=self.padding[2],
  64.                                    dilation=self.dilate[2])
  65.         out = self.bn2(out)
  66.         out += residual
  67.         out = self.act(out)
  68.  
  69.         return out

  71.     def forward(self, x):
  72.         xm = x
  73.         base_feat = []
  74.         if self.c is not False:

  76.             x2 = self.forward_for_middle(x)

  79.             x1 = self.forward_for_small(xm[0])

  83.         base_feat.append(x1)

  91.         super(RFEM, self).__init__()

  226. class MultiSEAM(nn.Module):
  227.     def __init__(self, c1, c2, depth, kernel_size=3, patch_size=[3, 5, 7], reduction=16):
  228.         super(MultiSEAM, self).__init__()
  229.         if c1 != c2:
  230.             c2 = c1
  231.         self.DCovN0 = DcovN(c1, c2, depth, kernel_size=kernel_size, patch_size=patch_size[0])
  232.         self.DCovN1 = DcovN(c1, c2, depth, kernel_size=kernel_size, patch_size=patch_size[1])
  233.         self.DCovN2 = DcovN(c1, c2, depth, kernel_size=kernel_size, patch_size=patch_size[2])
  234.         self.avg_pool = torch.nn.AdaptiveAvgPool2d(1)
  235.         self.fc = nn.Sequential(
  236.             nn.Linear(c2, c2 // reduction, bias=False),
  237.             nn.ReLU(inplace=True),
  238.             nn.Linear(c2 // reduction, c2, bias=False),
  239.             nn.Sigmoid()
  240.         )
  241.  
  242.     def forward(self, x):
  243.         b, c, _, _ = x.size()
  244.         y0 = self.DCovN0(x)
  245.         y1 = self.DCovN1(x)
  246.         y2 = self.DCovN2(x)
  247.         y0 = self.avg_pool(y0).view(b, c)
  248.         y1 = self.avg_pool(y1).view(b, c)
  249.         y2 = self.avg_pool(y2).view(b, c)
  250.         y4 = self.avg_pool(x).view(b, c)
  251.         y = (y0 + y1 + y2 + y4) / 4
  252.         y = self.fc(y).view(b, c, 1, 1)
  253.         y = torch.exp(y)
  254.         return x * y.expand_as(x)

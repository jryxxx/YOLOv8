1. SCConv介绍

 论文：https://openaccess.thecvf.com/content/CVPR2023/papers/Li_SCConv_Spatial_and_Channel_Reconstruction_Convolution_for_Feature_Redundancy_CVPR_2023_paper.pdf

         卷积神经网络（CNN）已经实现在各种计算机视觉任务中表现出色，但这是以巨大的计算成本为代价的资源，部分原因是卷积层提取冗余特征。 在本文中，我们尝试利用特征之间的空间和通道冗余，针对 CNN 压缩，提出了一种高效的卷积模块，称为 SCConv（空间和通道重建卷积），以减少冗余计算，并促进代表性特征学习。 提出的 SCConv 由两个单元组成：空间重建单元（SRU）和通道重建单元（CRU）。 SRU利用分离重建方法来抑制空间冗余，而CRU使用分割-变换-融合策略来减少通道冗余。 此外，SCConv 是一个即插即用的架构单元，可以可以直接用来替代各种卷积神经网络中的标准卷积。 实验结果表明SCConv 嵌入式模型能够实现更好的效果
通过减少冗余特征来显着降低复杂性和计算成本来提高性能。

        SCConv 的结构包括了空间重建单元（SRU）和通道重建单元（CRU）。 下图显示了我们的 SCConv 模块添加在 ResBlock 中的确切位置 。

  1. '''

  7. import torch
  8. import torch.nn.functional as F
  9. import torch.nn as nn
  10.  
  11. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  12.     # Pad to 'same' shape outputs
  13.     if d > 1:
  14.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  15.     if p is None:
  16.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  17.     return p
  18.  
  19.  
  20. class Conv(nn.Module):
  21.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  22.     default_act = nn.SiLU()  # default activation
  23.  
  24.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  25.         super().__init__()
  26.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  27.         self.bn = nn.BatchNorm2d(c2)
  28.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  29.  
  30.     def forward(self, x):
  31.         return self.act(self.bn(self.conv(x)))
  32.  
  33.     def forward_fuse(self, x):
  34.         return self.act(self.conv(x))
  35.  
  36.  
  37. class GroupBatchnorm2d(nn.Module):
  38.     def __init__(self, c_num:int, 
  39.                  group_num:int = 16, 
  40.                  eps:float = 1e-10
  41.                  ):
  42.         super(GroupBatchnorm2d,self).__init__()
  43.         assert c_num    >= group_num
  44.         self.group_num  = group_num
  45.         self.gamma      = nn.Parameter( torch.randn(c_num, 1, 1)    )
  46.         self.beta       = nn.Parameter( torch.zeros(c_num, 1, 1)    )
  47.         self.eps        = eps
  48.  
  49.     def forward(self, x):
  50.         N, C, H, W  = x.size()
  51.         x           = x.view(   N, self.group_num, -1   )
  52.         mean        = x.mean(   dim = 2, keepdim = True )
  53.         std         = x.std (   dim = 2, keepdim = True )
  54.         x           = (x - mean) / (std+self.eps)
  55.         x           = x.view(N, C, H, W)
  56.         return x * self.gamma + self.beta
  57.  
  58.  
  59. class SRU(nn.Module):
  60.     def __init__(self,
  61.                  oup_channels:int, 
  62.                  group_num:int = 16,
  63.                  gate_treshold:float = 0.5 
  64.                  ):
  65.         super().__init__()
  66.         
  67.         self.gn             = GroupBatchnorm2d( oup_channels, group_num = group_num )
  68.         self.gate_treshold  = gate_treshold
  69.         self.sigomid        = nn.Sigmoid()
  70.  
  71.     def forward(self,x):
  72.         gn_x        = self.gn(x)
  73.         w_gamma     = F.softmax(self.gn.gamma,dim=0)
  74.         reweigts    = self.sigomid( gn_x * w_gamma )
  75.         # Gate
  76.         info_mask   = w_gamma>self.gate_treshold
  77.         noninfo_mask= w_gamma<=self.gate_treshold
  78.         x_1         = info_mask*reweigts * x
  79.         x_2         = noninfo_mask*reweigts * x
  80.         x           = self.reconstruct(x_1,x_2)
  81.         return x
  82.     
  83.     def reconstruct(self,x_1,x_2):
  84.         x_11,x_12 = torch.split(x_1, x_1.size(1)//2, dim=1)
  85.         x_21,x_22 = torch.split(x_2, x_2.size(1)//2, dim=1)
  86.         return torch.cat([ x_11+x_22, x_12+x_21 ],dim=1)
  87.  
  88.  
  89. class CRU(nn.Module):
  90.     '''
  91.     alpha: 0<alpha<1
  92.     '''
  93.     def __init__(self, 
  94.                  op_channel:int,
  95.                  alpha:float = 1/2,
  96.                  squeeze_radio:int = 2 ,
  97.                  group_size:int = 2,
  98.                  group_kernel_size:int = 3,
  99.                  ):
  100.         super().__init__()
  101.         self.up_channel     = up_channel   =   int(alpha*op_channel)
  102.         self.low_channel    = low_channel  =   op_channel-up_channel
  103.         self.squeeze1       = nn.Conv2d(up_channel,up_channel//squeeze_radio,kernel_size=1,bias=False)
  104.         self.squeeze2       = nn.Conv2d(low_channel,low_channel//squeeze_radio,kernel_size=1,bias=False)
  105.         #up
  106.         self.GWC            = nn.Conv2d(up_channel//squeeze_radio, op_channel,kernel_size=group_kernel_size, stride=1,padding=group_kernel_size//2, groups = group_size)
  107.         self.PWC1           = nn.Conv2d(up_channel//squeeze_radio, op_channel,kernel_size=1, bias=False)
  108.         #low
  109.         self.PWC2           = nn.Conv2d(low_channel//squeeze_radio, op_channel-low_channel//squeeze_radio,kernel_size=1, bias=False)
  110.         self.advavg         = nn.AdaptiveAvgPool2d(1)
  111.  
  112.     def forward(self,x):
  113.         # Split
  114.         up,low  = torch.split(x,[self.up_channel,self.low_channel],dim=1)
  115.         up,low  = self.squeeze1(up),self.squeeze2(low)
  116.         # Transform
  117.         Y1      = self.GWC(up) + self.PWC1(up)
  118.         Y2      = torch.cat( [self.PWC2(low), low], dim= 1 )
  119.         # Fuse
  120.         out     = torch.cat( [Y1,Y2], dim= 1 )
  121.         out     = F.softmax( self.advavg(out), dim=1 ) * out
  122.         out1,out2 = torch.split(out,out.size(1)//2,dim=1)
  123.         return out1+out2
  124.  
  125.  
  126. class ScConv(nn.Module):
  127.     def __init__(self,
  128.                 op_channel:int,
  129.                 group_num:int = 16,
  130.                 gate_treshold:float = 0.5,
  131.                 alpha:float = 1/2,
  132.                 squeeze_radio:int = 2 ,
  133.                 group_size:int = 2,
  134.                 group_kernel_size:int = 3,
  135.                  ):
  136.         super().__init__()
  137.         self.SRU = SRU( op_channel, 
  138.                        group_num            = group_num,  
  139.                        gate_treshold        = gate_treshold )
  140.         self.CRU = CRU( op_channel, 
  141.                        alpha                = alpha, 
  142.                        squeeze_radio        = squeeze_radio ,
  143.                        group_size           = group_size ,
  144.                        group_kernel_size    = group_kernel_size )
  145.     
  146.     def forward(self,x):
  147.         x = self.SRU(x)
  148.         x = self.CRU(x)
  149.         return x
  150.  
  151.  
  152. class C2f_ScConv(nn.Module):
  153.     # CSP Bottleneck with 2 convolutions
  154.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  155.         super().__init__()
  156.         self.c = int(c2 * e)  # hidden channels
  157.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  158.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  159.         self.m = nn.ModuleList(ScConv(self.c) for _ in range(n))
  160.  
  161.     def forward(self, x):
  162.         y = list(self.cv1(x).chunk(2, 1))
  163.         y.extend(m(y[-1]) for m in self.m)
  164.         return self.cv2(torch.cat(y, 1))
  165.  
  166.     def forward_split(self, x):
  167.         y = list(self.cv1(x).split((self.c, self.c), 1))
  168.         y.extend(m(y[-1]) for m in self.m)
  169.         return self.cv2(torch.cat(y, 1))
  170.

  173.     model   = ScConv(32)

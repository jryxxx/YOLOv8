  11. class DoubleAttention(nn.Module):
  12.  
  13.     def __init__(self, in_channels,c_m=128,c_n=128,reconstruct = True):
  14.         super().__init__()
  15.         self.in_channels=in_channels
  16.         self.reconstruct = reconstruct
  17.         self.c_m=c_m
  18.         self.c_n=c_n
  19.         self.convA=nn.Conv2d(in_channels,c_m,1)
  20.         self.convB=nn.Conv2d(in_channels,c_n,1)
  21.         self.convV=nn.Conv2d(in_channels,c_n,1)
  22.         if self.reconstruct:
  23.             self.conv_reconstruct = nn.Conv2d(c_m, in_channels, kernel_size = 1)
  24.         self.init_weights()
  25.  
  26.  
  27.     def init_weights(self):
  28.         for m in self.modules():
  29.             if isinstance(m, nn.Conv2d):
  30.                 init.kaiming_normal_(m.weight, mode='fan_out')
  31.                 if m.bias is not None:
  32.                     init.constant_(m.bias, 0)
  33.             elif isinstance(m, nn.BatchNorm2d):
  34.                 init.constant_(m.weight, 1)
  35.                 init.constant_(m.bias, 0)
  36.             elif isinstance(m, nn.Linear):
  37.                 init.normal_(m.weight, std=0.001)
  38.                 if m.bias is not None:
  39.                     init.constant_(m.bias, 0)
  40.  
  41.     def forward(self, x):
  42.         b, c, h,w=x.shape
  43.         assert c==self.in_channels
  44.         A=self.convA(x) #b,c_m,h,w
  45.         B=self.convB(x) #b,c_n,h,w
  46.         V=self.convV(x) #b,c_n,h,w
  47.         tmpA=A.view(b,self.c_m,-1)
  48.         attention_maps=F.softmax(B.view(b,self.c_n,-1))
  49.         attention_vectors=F.softmax(V.view(b,self.c_n,-1))
  50.         # step 1: feature gating
  51.         global_descriptors=torch.bmm(tmpA,attention_maps.permute(0,2,1)) #b.c_m,c_n
  52.         # step 2: feature distribution
  53.         tmpZ = global_descriptors.matmul(attention_vectors) #b,c_m,h*w
  54.         tmpZ=tmpZ.view(b,self.c_m,h,w) #b,c_m,h,w
  55.         if self.reconstruct:
  56.             tmpZ=self.conv_reconstruct(tmpZ)
  57.  
  58.         return tmpZ

  3. from torch.nn import init
  4. from collections import OrderedDict
  5.  
  6.  
  7. class SKAttention(nn.Module):
  8.  
  9.     def __init__(self, channel=512, kernels=[1, 3, 5, 7], reduction=16, group=1, L=32):
  10.         super().__init__()
  11.         self.d = max(L, channel // reduction)
  12.         self.convs = nn.ModuleList([])
  13.         for k in kernels:
  14.             self.convs.append(
  15.                 nn.Sequential(OrderedDict([
  16.                     ('conv', nn.Conv2d(channel, channel, kernel_size=k, padding=k // 2, groups=group)),
  17.                     ('bn', nn.BatchNorm2d(channel)),
  18.                     ('relu', nn.ReLU())
  19.                 ]))
  20.             )
  21.         self.fc = nn.Linear(channel, self.d)
  22.         self.fcs = nn.ModuleList([])
  23.         for i in range(len(kernels)):
  24.             self.fcs.append(nn.Linear(self.d, channel))
  25.         self.softmax = nn.Softmax(dim=0)
  26.  
  27.     def forward(self, x):
  28.         bs, c, _, _ = x.size()
  29.         conv_outs = []
  30.         ### split
  31.         for conv in self.convs:
  32.             conv_outs.append(conv(x))
  33.         feats = torch.stack(conv_outs, 0)  # k,bs,channel,h,w
  34.  
  35.         ### fuse
  36.         U = sum(conv_outs)  # bs,c,h,w
  37.  
  38.         ### reduction channel
  39.         S = U.mean(-1).mean(-1)  # bs,c
  40.         Z = self.fc(S)  # bs,d
  41.  
  42.         ### calculate attention weight
  43.         weights = []
  44.         for fc in self.fcs:
  45.             weight = fc(Z)
  46.             weights.append(weight.view(bs, c, 1, 1))  # bs,channel
  47.         attention_weughts = torch.stack(weights, 0)  # k,bs,channel,1,1
  48.         attention_weughts = self.softmax(attention_weughts)  # k,bs,channel,1,1
  49.  
  50.         ### fuse
  51.         V = (attention_weughts * feats).sum(0)
  52.         return V
  53.

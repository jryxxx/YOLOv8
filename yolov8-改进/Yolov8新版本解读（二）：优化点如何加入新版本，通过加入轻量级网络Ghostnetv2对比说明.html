  3. """
  4. Creates a GhostNet Model as defined in:
  5. GhostNet: More Features from Cheap Operations By Kai Han, Yunhe Wang, Qi Tian, Jianyuan Guo, Chunjing Xu, Chang Xu.
  6. https://arxiv.org/abs/1911.11907
  7. Modified from https://github.com/d-li14/mobilenetv3.pytorch and https://github.com/rwightman/pytorch-image-models
  8. """
  9. import torch
  10. import torch.nn as nn
  11. import torch.nn.functional as F
  12. import math
  13.  
  14. from timm.models.registry import register_model
  15.

  18. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  19.     # Pad to 'same' shape outputs
  20.     if d > 1:
  21.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  22.     if p is None:
  23.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  24.     return p
  25.  
  26.  
  27. class Conv(nn.Module):
  28.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  29.     default_act = nn.SiLU()  # default activation
  30.  
  31.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  32.         super().__init__()
  33.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  34.         self.bn = nn.BatchNorm2d(c2)
  35.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  36.  
  37.     def forward(self, x):
  38.         return self.act(self.bn(self.conv(x)))
  39.  
  40.     def forward_fuse(self, x):
  41.         return self.act(self.conv(x))
  42.  
  43. def _make_divisible(v, divisor, min_value=None):
  44.     """
  45.     This function is taken from the original tf repo.
  46.     It ensures that all layers have a channel number that is divisible by 8
  47.     It can be seen here:
  48.     https://github.com/tensorflow/models/blob/master/research/slim/nets/mobilenet/mobilenet.py
  49.     """
  50.     if min_value is None:
  51.         min_value = divisor
  52.     new_v = max(min_value, int(v + divisor / 2) // divisor * divisor)
  53.     # Make sure that round down does not go down by more than 10%.
  54.     if new_v < 0.9 * v:
  55.         new_v += divisor
  56.     return new_v
  57.  
  58. def hard_sigmoid(x, inplace: bool = False):
  59.     if inplace:
  60.         return x.add_(3.).clamp_(0., 6.).div_(6.)
  61.     else:
  62.         return F.relu6(x + 3.) / 6.
  63.  
  64. class SqueezeExcite(nn.Module):
  65.     def __init__(self, in_chs, se_ratio=0.25, reduced_base_chs=None,
  66.                  act_layer=nn.ReLU, gate_fn=hard_sigmoid, divisor=4, **_):
  67.         super(SqueezeExcite, self).__init__()
  68.         self.gate_fn = gate_fn
  69.         reduced_chs = _make_divisible((reduced_base_chs or in_chs) * se_ratio, divisor)
  70.         self.avg_pool = nn.AdaptiveAvgPool2d(1)
  71.         self.conv_reduce = nn.Conv2d(in_chs, reduced_chs, 1, bias=True)
  72.         self.act1 = act_layer(inplace=True)
  73.         self.conv_expand = nn.Conv2d(reduced_chs, in_chs, 1, bias=True)
  74.  
  75.     def forward(self, x):
  76.         x_se = self.avg_pool(x)
  77.         x_se = self.conv_reduce(x_se)
  78.         x_se = self.act1(x_se)
  79.         x_se = self.conv_expand(x_se)
  80.         x = x * self.gate_fn(x_se)
  81.         return x    
  82.   
  83. class ConvBnAct(nn.Module):
  84.     def __init__(self, in_chs, out_chs, kernel_size,
  85.                  stride=1, act_layer=nn.ReLU):
  86.         super(ConvBnAct, self).__init__()
  87.         self.conv = nn.Conv2d(in_chs, out_chs, kernel_size, stride, kernel_size//2, bias=False)
  88.         self.bn1 = nn.BatchNorm2d(out_chs)
  89.         self.act1 = act_layer(inplace=True)
  90.  
  91.     def forward(self, x):
  92.         x = self.conv(x)
  93.         x = self.bn1(x)
  94.         x = self.act1(x)
  95.         return x

  98. class GhostModuleV2(nn.Module):
  99.     def __init__(self, inp, oup, kernel_size=1, ratio=2, dw_size=3, stride=1, relu=True, mode=None, args=None):
  100.         super(GhostModuleV2, self).__init__()
  101.         self.mode = mode
  102.         self.gate_fn = nn.Sigmoid()
  103.  
  104.         if self.mode in ['original']:
  105.             self.oup = oup
  106.             init_channels = math.ceil(oup / ratio)
  107.             new_channels = init_channels * (ratio - 1)
  108.             self.primary_conv = nn.Sequential(
  109.                 nn.Conv2d(inp, init_channels, kernel_size, stride, kernel_size // 2, bias=False),
  110.                 nn.BatchNorm2d(init_channels),
  111.                 nn.ReLU(inplace=True) if relu else nn.Sequential(),
  112.             )
  113.             self.cheap_operation = nn.Sequential(
  114.                 nn.Conv2d(init_channels, new_channels, dw_size, 1, dw_size // 2, groups=init_channels, bias=False),
  115.                 nn.BatchNorm2d(new_channels),
  116.                 nn.ReLU(inplace=True) if relu else nn.Sequential(),
  117.             )
  118.         elif self.mode in ['attn']:
  119.             self.oup = oup
  120.             init_channels = math.ceil(oup / ratio)
  121.             new_channels = init_channels * (ratio - 1)
  122.             self.primary_conv = nn.Sequential(
  123.                 nn.Conv2d(inp, init_channels, kernel_size, stride, kernel_size // 2, bias=False),
  124.                 nn.BatchNorm2d(init_channels),
  125.                 nn.ReLU(inplace=True) if relu else nn.Sequential(),
  126.             )
  127.             self.cheap_operation = nn.Sequential(
  128.                 nn.Conv2d(init_channels, new_channels, dw_size, 1, dw_size // 2, groups=init_channels, bias=False),
  129.                 nn.BatchNorm2d(new_channels),
  130.                 nn.ReLU(inplace=True) if relu else nn.Sequential(),
  131.             )
  132.             self.short_conv = nn.Sequential(
  133.                 nn.Conv2d(inp, oup, kernel_size, stride, kernel_size // 2, bias=False),
  134.                 nn.BatchNorm2d(oup),
  135.                 nn.Conv2d(oup, oup, kernel_size=(1, 5), stride=1, padding=(0, 2), groups=oup, bias=False),
  136.                 nn.BatchNorm2d(oup),
  137.                 nn.Conv2d(oup, oup, kernel_size=(5, 1), stride=1, padding=(2, 0), groups=oup, bias=False),
  138.                 nn.BatchNorm2d(oup),
  139.             )
  140.  
  141.     def forward(self, x):
  142.         if self.mode in ['original']:
  143.             x1 = self.primary_conv(x)
  144.             x2 = self.cheap_operation(x1)
  145.             out = torch.cat([x1, x2], dim=1)
  146.             return out[:, :self.oup, :, :]
  147.         elif self.mode in ['attn']:
  148.             res = self.short_conv(F.avg_pool2d(x, kernel_size=2, stride=2))
  149.             x1 = self.primary_conv(x)
  150.             x2 = self.cheap_operation(x1)
  151.             out = torch.cat([x1, x2], dim=1)
  152.             return out[:, :self.oup, :, :] * F.interpolate(self.gate_fn(res), size=(out.shape[-2], out.shape[-1]),
  153.                                                            mode='nearest')
  154.

  156. class GhostBottleneckV2(nn.Module):
  157.  
  158.     def __init__(self, in_chs, mid_chs, out_chs, dw_kernel_size=3,
  159.                  stride=1, act_layer=nn.ReLU, se_ratio=0., layer_id=0, args=None):
  160.         super(GhostBottleneckV2, self).__init__()
  161.         has_se = se_ratio is not None and se_ratio > 0.
  162.         self.stride = stride
  163.  
  164.  
  165.         # Point-wise expansion
  166.         if layer_id <= 1:
  167.             self.ghost1 = GhostModuleV2(in_chs, mid_chs, relu=True, mode='original', args=args)
  168.         else:
  169.             self.ghost1 = GhostModuleV2(in_chs, mid_chs, relu=True, mode='attn', args=args)
  170.  
  171.             # Depth-wise convolution
  172.         if self.stride > 1:
  173.             self.conv_dw = nn.Conv2d(mid_chs, mid_chs, dw_kernel_size, stride=stride,
  174.                                      padding=(dw_kernel_size - 1) // 2, groups=mid_chs, bias=False)
  175.             self.bn_dw = nn.BatchNorm2d(mid_chs)
  176.  
  177.         # Squeeze-and-excitation
  178.         if has_se:
  179.             self.se = SqueezeExcite(mid_chs, se_ratio=se_ratio)
  180.         else:
  181.             self.se = None
  182.  
  183.         self.ghost2 = GhostModuleV2(mid_chs, out_chs, relu=False, mode='original', args=args)
  184.  
  185.         # shortcut
  186.         if (in_chs == out_chs and self.stride == 1):
  187.             self.shortcut = nn.Sequential()
  188.         else:
  189.             self.shortcut = nn.Sequential(
  190.                 nn.Conv2d(in_chs, in_chs, dw_kernel_size, stride=stride,
  191.                           padding=(dw_kernel_size - 1) // 2, groups=in_chs, bias=False),
  192.                 nn.BatchNorm2d(in_chs),
  193.                 nn.Conv2d(in_chs, out_chs, 1, stride=1, padding=0, bias=False),
  194.                 nn.BatchNorm2d(out_chs),
  195.             )
  196.  
  197.     def forward(self, x):
  198.         residual = x
  199.         x = self.ghost1(x)
  200.         if self.stride > 1:
  201.             x = self.conv_dw(x)
  202.             x = self.bn_dw(x)
  203.         if self.se is not None:
  204.             x = self.se(x)
  205.         x = self.ghost2(x)
  206.         x += self.shortcut(residual)
  207.         return x
  208.

  210. class C2f_GhostBottleneckV2(nn.Module):
  211.     # CSP Bottleneck with 2 convolutions
  212.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  213.         super().__init__()
  214.         self.c = int(c2 * e)  # hidden channels
  215.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  216.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  217.         self.m = nn.ModuleList(GhostBottleneckV2(self.c, self.c, self.c) for _ in range(n))
  218.  
  219.     def forward(self, x):
  220.         y = list(self.cv1(x).chunk(2, 1))
  221.         y.extend(m(y[-1]) for m in self.m)
  222.         return self.cv2(torch.cat(y, 1))
  223.  
  224.     def forward_split(self, x):
  225.         y = list(self.cv1(x).split((self.c, self.c), 1))
  226.         y.extend(m(y[-1]) for m in self.m)
  227.         return self.cv2(torch.cat(y, 1))

  229. class GhostNetV2(nn.Module):
  230.     def __init__(self, cfgs, num_classes=1000, width=1.0, dropout=0.2,block=GhostBottleneckV2,args=None):
  231.         super(GhostNetV2, self).__init__()
  232.         self.cfgs = cfgs
  233.         self.dropout = dropout
  234.  
  235.         # building first layer
  236.         output_channel = _make_divisible(16 * width, 4)
  237.         self.conv_stem = nn.Conv2d(3, output_channel, 3, 2, 1, bias=False)
  238.         self.bn1 = nn.BatchNorm2d(output_channel)
  239.         self.act1 = nn.ReLU(inplace=True)
  240.         input_channel = output_channel
  241.  
  242.         # building inverted residual blocks
  243.         stages = []
  244.         #block = block
  245.         layer_id=0
  246.         for cfg in self.cfgs:
  247.             layers = []
  248.             for k, exp_size, c, se_ratio, s in cfg:
  249.                 output_channel = _make_divisible(c * width, 4)
  250.                 hidden_channel = _make_divisible(exp_size * width, 4)
  251.                 if block==GhostBottleneckV2:
  252.                     layers.append(block(input_channel, hidden_channel, output_channel, k, s,
  253.                                   se_ratio=se_ratio,layer_id=layer_id,args=args))
  254.                 input_channel = output_channel
  255.                 layer_id+=1
  256.             stages.append(nn.Sequential(*layers))
  257.  
  258.         output_channel = _make_divisible(exp_size * width, 4)
  259.         stages.append(nn.Sequential(ConvBnAct(input_channel, output_channel, 1)))
  260.         input_channel = output_channel
  261.         
  262.         self.blocks = nn.Sequential(*stages)        
  263.  
  264.         # building last several layers
  265.         output_channel = 1280
  266.         self.global_pool = nn.AdaptiveAvgPool2d((1, 1))
  267.         self.conv_head = nn.Conv2d(input_channel, output_channel, 1, 1, 0, bias=True)
  268.         self.act2 = nn.ReLU(inplace=True)
  269.         self.classifier = nn.Linear(output_channel, num_classes)
  270.  
  271.     def forward(self, x):
  272.         x = self.conv_stem(x)
  273.         x = self.bn1(x)
  274.         x = self.act1(x)
  275.         x = self.blocks(x)
  276.         x = self.global_pool(x)
  277.         x = self.conv_head(x)
  278.         x = self.act2(x)
  279.         x = x.view(x.size(0), -1)
  280.         if self.dropout > 0.:
  281.             x = F.dropout(x, p=self.dropout, training=self.training)
  282.         x = self.classifier(x)
  283.         return x
  284.

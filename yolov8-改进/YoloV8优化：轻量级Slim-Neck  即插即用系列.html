  3. class GSConv(nn.Module):
  4.     # GSConv https://github.com/AlanLi1997/slim-neck-by-gsconv
  5.     def __init__(self, c1, c2, k=1, s=1, g=1, act=True):
  6.         super().__init__()
  7.         c_ = c2 // 2
  8.         self.cv1 = Conv(c1, c_, k, s, None, g, 1,  act)
  9.         self.cv2 = Conv(c_, c_, 5, 1, None, c_, 1 , act)
  10.  
  11.     def forward(self, x):
  12.         x1 = self.cv1(x)
  13.         x2 = torch.cat((x1, self.cv2(x1)), 1)
  14.         # shuffle
  15.         # y = x2.reshape(x2.shape[0], 2, x2.shape[1] // 2, x2.shape[2], x2.shape[3])
  16.         # y = y.permute(0, 2, 1, 3, 4)
  17.         # return y.reshape(y.shape[0], -1, y.shape[3], y.shape[4])
  18.  
  19.         b, n, h, w = x2.data.size()
  20.         b_n = b * n // 2
  21.         y = x2.reshape(b_n, 2, h * w)
  22.         y = y.permute(1, 0, 2)
  23.         y = y.reshape(2, -1, n // 2, h, w)
  24.  
  25.         return torch.cat((y[0], y[1]), 1)
  26.  
  27. class GSConvns(GSConv):
  28.     # GSConv with a normative-shuffle https://github.com/AlanLi1997/slim-neck-by-gsconv
  29.     def __init__(self, c1, c2, k=1, s=1, g=1, act=True):
  30.         super().__init__(c1, c2, k=1, s=1, g=1, act=True)
  31.         c_ = c2 // 2
  32.         self.shuf = nn.Conv2d(c_ * 2, c2, 1, 1, 0, bias=False)
  33.  
  34.     def forward(self, x):
  35.         x1 = self.cv1(x)
  36.         x2 = torch.cat((x1, self.cv2(x1)), 1)
  37.         # normative-shuffle, TRT supported
  38.         return nn.ReLU(self.shuf(x2))
  39.

  41. class GSBottleneck(nn.Module):
  42.     # GS Bottleneck https://github.com/AlanLi1997/slim-neck-by-gsconv
  43.     def __init__(self, c1, c2, k=3, s=1, e=0.5):
  44.         super().__init__()
  45.         c_ = int(c2*e)
  46.         # for lighting
  47.         self.conv_lighting = nn.Sequential(
  48.             GSConv(c1, c_, 1, 1),
  49.             GSConv(c_, c2, 3, 1, act=False))
  50.         self.shortcut = Conv(c1, c2, 1, 1, act=False)
  51.  
  52.     def forward(self, x):
  53.         return self.conv_lighting(x) + self.shortcut(x)
  54.

  56. class DWConv(Conv):
  57.     # Depth-wise convolution class
  58.     def __init__(self, c1, c2, k=1, s=1, act=True):  # ch_in, ch_out, kernel, stride, padding, groups
  59.         super().__init__(c1, c2, k, s, g=math.gcd(c1, c2), act=act)
  60.

  62. class GSBottleneckC(GSBottleneck):
  63.     # cheap GS Bottleneck https://github.com/AlanLi1997/slim-neck-by-gsconv
  64.     def __init__(self, c1, c2, k=3, s=1):
  65.         super().__init__(c1, c2, k, s)
  66.         self.shortcut = DWConv(c1, c2, k, s, act=False)
  67.

  69. class VoVGSCSP(nn.Module):
  70.     # VoVGSCSP module with GSBottleneck
  71.     def __init__(self, c1, c2, n=1, shortcut=True, g=1, e=0.5):
  72.         super().__init__()
  73.         c_ = int(c2 * e)  # hidden channels
  74.         self.cv1 = Conv(c1, c_, 1, 1)
  75.         self.cv2 = Conv(c1, c_, 1, 1)
  76.         # self.gc1 = GSConv(c_, c_, 1, 1)
  77.         # self.gc2 = GSConv(c_, c_, 1, 1)
  78.         # self.gsb = GSBottleneck(c_, c_, 1, 1)
  79.         self.gsb = nn.Sequential(*(GSBottleneck(c_, c_, e=1.0) for _ in range(n)))
  80.         self.res = Conv(c_, c_, 3, 1, act=False)
  81.         self.cv3 = Conv(2 * c_, c2, 1)  #
  82.  
  83.  
  84.     def forward(self, x):
  85.         x1 = self.gsb(self.cv1(x))
  86.         y = self.cv2(x)
  87.         return self.cv3(torch.cat((y, x1), dim=1))
  88.  
  89.  
  90. class VoVGSCSPC(VoVGSCSP):
  91.     # cheap VoVGSCSP module with GSBottleneck
  92.     def __init__(self, c1, c2, n=1, shortcut=True, g=1, e=0.5):
  93.         super().__init__(c1, c2)
  94.         c_ = int(c2 * 0.5)  # hidden channels
  95.         self.gsb = GSBottleneckC(c_, c_, 1, 1)

  1. class BasicRFB(nn.Module):
  2.  
  3.     def __init__(self, in_planes, out_planes, stride=1, scale=0.1, map_reduce=8, vision=1, groups=1):
  4.         super(BasicRFB, self).__init__()
  5.         self.scale = scale
  6.         self.out_channels = out_planes
  7.         inter_planes = in_planes // map_reduce
  8.  
  9.         self.branch0 = nn.Sequential(
  10.             BasicConv(in_planes, inter_planes, kernel_size=1, stride=1, groups=groups, relu=False),
  11.             BasicConv(inter_planes, 2 * inter_planes, kernel_size=(3, 3), stride=stride, padding=(1, 1), groups=groups),
  12.             BasicConv(2 * inter_planes, 2 * inter_planes, kernel_size=3, stride=1, padding=vision + 1,
  13.                       dilation=vision + 1, relu=False, groups=groups)
  14.         )
  15.         self.branch1 = nn.Sequential(
  16.             BasicConv(in_planes, inter_planes, kernel_size=1, stride=1, groups=groups, relu=False),
  17.             BasicConv(inter_planes, 2 * inter_planes, kernel_size=(3, 3), stride=stride, padding=(1, 1), groups=groups),
  18.             BasicConv(2 * inter_planes, 2 * inter_planes, kernel_size=3, stride=1, padding=vision + 2,
  19.                       dilation=vision + 2, relu=False, groups=groups)
  20.         )
  21.         self.branch2 = nn.Sequential(
  22.             BasicConv(in_planes, inter_planes, kernel_size=1, stride=1, groups=groups, relu=False),
  23.             BasicConv(inter_planes, (inter_planes // 2) * 3, kernel_size=3, stride=1, padding=1, groups=groups),
  24.             BasicConv((inter_planes // 2) * 3, 2 * inter_planes, kernel_size=3, stride=stride, padding=1,
  25.                       groups=groups),
  26.             BasicConv(2 * inter_planes, 2 * inter_planes, kernel_size=3, stride=1, padding=vision + 4,
  27.                       dilation=vision + 4, relu=False, groups=groups)
  28.         )
  29.  
  30.         self.ConvLinear = BasicConv(6 * inter_planes, out_planes, kernel_size=1, stride=1, relu=False)
  31.         self.shortcut = BasicConv(in_planes, out_planes, kernel_size=1, stride=stride, relu=False)
  32.         self.relu = nn.ReLU(inplace=False)
  33.  
  34.     def forward(self, x):
  35.         x0 = self.branch0(x)
  36.         x1 = self.branch1(x)
  37.         x2 = self.branch2(x)
  38.  
  39.         out = torch.cat((x0, x1, x2), 1)
  40.         out = self.ConvLinear(out)
  41.         short = self.shortcut(x)
  42.         out = out * self.scale + short
  43.         out = self.relu(out)
  44.  
  45.         return out

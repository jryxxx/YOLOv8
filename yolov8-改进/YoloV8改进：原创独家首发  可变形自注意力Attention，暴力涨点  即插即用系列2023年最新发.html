 1.Deformable Attention Transformer

CVPR 2022 视觉领域顶级学术会议上，该项工作进入了 Best Paper 奖项的候选角逐。 

 本文提出了一种简单有效的可变形的自注意力模块，并在此模块上构造了一个强大的Pyramid Backbone，即可变形的注意力Transformer(Deformable Attention Transformer, DAT)，用于图像分类和各种密集的预测任务。研究者们让所有 query 都跟同一组 key 和 value 交互，通过对每个输⼊图像学习⼀组偏移量，移动 key 和 value 到重要的位置。这种设计不仅增强了 sparse attention 的表征能⼒，同时具有线性空间复杂度。

         可变形注意机制的图示。 (a) 呈现可变形注意力的信息流。在左侧部分，一组参考点均匀地放置在特征图上，其偏移量是通过偏移网络从查询中学习到的。然后根据变形点从采样的特征中投影出变形的键和值，如右图所示。相对位置偏差也由变形点计算，增强了输出变形特征的多头注意力。我们只展示了 4 个参考点以进行清晰的展示，实际实施中还有更多参考点。 (b) 揭示了偏移生

与 DCN 类似，将设计好的可变形注意力模块加入到 ViT 模型的最后两个阶段，得到 Deformable Attention Transformer，如下图所示：

  • 在分类任务中，首先对最后一阶段输出的特征图进行归一化处理，然后采用具有合并特征的线性分类器来预测logits。
  • 在目标检测、实例分割和语义分割任务中，DAT扮演着Backbone的作用，以提取多尺度特征。

  1. import torch
  2. import torch.nn.functional as F
  3. from torch import nn, einsum
  4.  
  5. from einops import rearrange, repeat
  6.  
  7. # helper functions
  8.  
  9. def exists(val):
  10.     return val is not None
  11.  
  12. def default(val, d):
  13.     return val if exists(val) else d
  14.  
  15. def divisible_by(numer, denom):
  16.     return (numer % denom) == 0
  17.  
  18. # tensor helpers
  19.  
  20. def create_grid_like(t, dim = 0):
  21.     h, w, device = *t.shape[-2:], t.device
  22.  
  23.     grid = torch.stack(torch.meshgrid(
  24.         torch.arange(h, device = device),
  25.         torch.arange(w, device = device),
  26.     indexing = 'ij'), dim = dim)
  27.  
  28.     grid.requires_grad = False
  29.     grid = grid.type_as(t)
  30.     return grid
  31.  
  32. def normalize_grid(grid, dim = 1, out_dim = -1):
  33.     # normalizes a grid to range from -1 to 1
  34.     h, w = grid.shape[-2:]
  35.     grid_h, grid_w = grid.unbind(dim = dim)
  36.  
  37.     grid_h = 2.0 * grid_h / max(h - 1, 1) - 1.0
  38.     grid_w = 2.0 * grid_w / max(w - 1, 1) - 1.0
  39.  
  40.     return torch.stack((grid_h, grid_w), dim = out_dim)
  41.  
  42. class Scale(nn.Module):
  43.     def __init__(self, scale):
  44.         super().__init__()
  45.         self.scale = scale
  46.  
  47.     def forward(self, x):
  48.         return x * self.scale
  49.  
  50. # continuous positional bias from SwinV2
  51.  
  52. class CPB(nn.Module):
  53.     """ https://arxiv.org/abs/2111.09883v1 """
  54.  
  55.     def __init__(self, dim, heads, offset_groups, depth):
  56.         super().__init__()
  57.         self.heads = heads
  58.         self.offset_groups = offset_groups
  59.  
  60.         self.mlp = nn.ModuleList([])
  61.  
  62.         self.mlp.append(nn.Sequential(
  63.             nn.Linear(2, dim),
  64.             nn.ReLU()
  65.         ))
  66.  
  67.         for _ in range(depth - 1):
  68.             self.mlp.append(nn.Sequential(
  69.                 nn.Linear(dim, dim),
  70.                 nn.ReLU()
  71.             ))
  72.  
  73.         self.mlp.append(nn.Linear(dim, heads // offset_groups))
  74.  
  75.     def forward(self, grid_q, grid_kv):
  76.         device, dtype = grid_q.device, grid_kv.dtype
  77.  
  78.         grid_q = rearrange(grid_q, 'h w c -> 1 (h w) c')
  79.         grid_kv = rearrange(grid_kv, 'b h w c -> b (h w) c')
  80.  
  81.         pos = rearrange(grid_q, 'b i c -> b i 1 c') - rearrange(grid_kv, 'b j c -> b 1 j c')
  82.         bias = torch.sign(pos) * torch.log(pos.abs() + 1)  # log of distance is sign(rel_pos) * log(abs(rel_pos) + 1)
  83.  
  84.         for layer in self.mlp:
  85.             bias = layer(bias)
  86.  
  87.         bias = rearrange(bias, '(b g) i j o -> b (g o) i j', g = self.offset_groups)
  88.  
  89.         return bias
  90.  
  91. # main class
  92.  
  93. class DeformableAttention2D(nn.Module):
  94.     def __init__(
  95.         self,
  96.         dim,
  97.         dim_head = 64,
  98.         heads = 8,
  99.         dropout = 0.,
  100.         downsample_factor = 4,
  101.         offset_scale = None,
  102.         offset_groups = None,
  103.         offset_kernel_size = 6,
  104.         group_queries = True,
  105.         group_key_values = True
  106.     ):
  107.         super().__init__()
  108.         offset_scale = default(offset_scale, downsample_factor)
  109.         assert offset_kernel_size >= downsample_factor, 'offset kernel size must be greater than or equal to the downsample factor'
  110.         assert divisible_by(offset_kernel_size - downsample_factor, 2)
  111.  
  112.         offset_groups = default(offset_groups, heads)
  113.         assert divisible_by(heads, offset_groups)
  114.  
  115.         inner_dim = dim_head * heads
  116.         self.scale = dim_head ** -0.5
  117.         self.heads = heads
  118.         self.offset_groups = offset_groups
  119.  
  120.         offset_dims = inner_dim // offset_groups
  121.  
  122.         self.downsample_factor = downsample_factor
  123.  
  124.         self.to_offsets = nn.Sequential(
  125.             nn.Conv2d(offset_dims, offset_dims, offset_kernel_size, groups = offset_dims, stride = downsample_factor, padding = (offset_kernel_size - downsample_factor) // 2),
  126.             nn.GELU(),
  127.             nn.Conv2d(offset_dims, 2, 1, bias = False),
  128.             nn.Tanh(),
  129.             Scale(offset_scale)
  130.         )
  131.  
  132.         self.rel_pos_bias = CPB(dim // 4, offset_groups = offset_groups, heads = heads, depth = 2)
  133.  
  134.         self.dropout = nn.Dropout(dropout)
  135.         self.to_q = nn.Conv2d(dim, inner_dim, 1, groups = offset_groups if group_queries else 1, bias = False)
  136.         self.to_k = nn.Conv2d(dim, inner_dim, 1, groups = offset_groups if group_key_values else 1, bias = False)
  137.         self.to_v = nn.Conv2d(dim, inner_dim, 1, groups = offset_groups if group_key_values else 1, bias = False)
  138.         self.to_out = nn.Conv2d(inner_dim, dim, 1)
  139.  
  140.     def forward(self, x, return_vgrid = False):
  141.         """
  142.         b - batch
  143.         h - heads
  144.         x - height
  145.         y - width
  146.         d - dimension
  147.         g - offset groups
  148.         """
  149.  
  150.         heads, b, h, w, downsample_factor, device = self.heads, x.shape[0], *x.shape[-2:], self.downsample_factor, x.device
  151.  
  152.         # queries
  153.  
  154.         q = self.to_q(x)
  155.  
  156.         # calculate offsets - offset MLP shared across all groups
  157.  
  158.         group = lambda t: rearrange(t, 'b (g d) ... -> (b g) d ...', g = self.offset_groups)
  159.  
  160.         grouped_queries = group(q)
  161.         offsets = self.to_offsets(grouped_queries)
  162.  
  163.         # calculate grid + offsets
  164.  
  165.         grid =create_grid_like(offsets)
  166.         vgrid = grid + offsets
  167.  
  168.         vgrid_scaled = normalize_grid(vgrid)
  169.  
  170.         kv_feats = F.grid_sample(
  171.             group(x),
  172.             vgrid_scaled,
  173.         mode = 'bilinear', padding_mode = 'zeros', align_corners = False)
  174.  
  175.         kv_feats = rearrange(kv_feats, '(b g) d ... -> b (g d) ...', b = b)
  176.  
  177.         # derive key / values
  178.  
  179.         k, v = self.to_k(kv_feats), self.to_v(kv_feats)
  180.  
  181.         # scale queries
  182.  
  183.         q = q * self.scale
  184.  
  185.         # split out heads
  186.  
  187.         q, k, v = map(lambda t: rearrange(t, 'b (h d) ... -> b h (...) d', h = heads), (q, k, v))
  188.  
  189.         # query / key similarity
  190.  
  191.         sim = einsum('b h i d, b h j d -> b h i j', q, k)
  192.  
  193.         # relative positional bias
  194.  
  195.         grid = create_grid_like(x)
  196.         grid_scaled = normalize_grid(grid, dim = 0)
  197.         rel_pos_bias = self.rel_pos_bias(grid_scaled, vgrid_scaled)
  198.         sim = sim + rel_pos_bias
  199.  
  200.         # numerical stability
  201.  
  202.         sim = sim - sim.amax(dim = -1, keepdim = True).detach()
  203.  
  204.         # attention
  205.  
  206.         attn = sim.softmax(dim = -1)
  207.         attn = self.dropout(attn)
  208.  
  209.         # aggregate and combine heads
  210.  
  211.         out = einsum('b h i j, b h j d -> b h i d', attn, v)
  212.         out = rearrange(out, 'b h (x y) d -> b (h d) x y', x = h, y = w)
  213.         out = self.to_out(out)
  214.  
  215.         if return_vgrid:
  216.             return out, vgrid
  217.  
  218.         return out
  219.

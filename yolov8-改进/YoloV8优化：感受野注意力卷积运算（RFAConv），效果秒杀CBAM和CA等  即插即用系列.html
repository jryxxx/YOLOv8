 1.RFAConv介绍

 论文：https://arxiv.org/pdf/2304.03198.pdf

摘要：空间注意力已被广泛用于提高卷积神经网络的性能。 然而，它有一定的局限性。 在本文中，我们提出了空间注意力有效性的新视角，即空间注意力机制本质上解决了卷积核参数共享的问题。 然而，空间注意力生成的注意力图中包含的信息对于大尺寸的卷积核来说是不够的。 因此，我们提出了一种新颖的注意力机制，称为感受野注意力（RFA）。 现有的空间注意力，例如卷积块注意力模块（CBAM）和协调注意力（CA）仅关注空间特征，并没有完全解决卷积核参数共享的问题。 相比之下，RFA 不仅关注感受野空间特征，还为大尺寸卷积核提供有效的注意力权重。 RFA 开发的感受野注意力卷积运算（RFAConv）代表了一种替代标准卷积运算的新方法。 它提供的计算成本和参数增量几乎可以忽略不计，同时显着提高了网络性能。 我们在 ImageNet-1k、COCO 和 VOC 数据集上进行了一系列实验，以证明我们方法的优越性。 特别重要的是，我们认为现在是时候将当前空间注意机制的焦点从空间特征转移到感受野空间特征了。 这样，我们就可以进一步提升网络性能，取得更好的效果。

        关于感受野空间特征，我们提出感受野注意（RFA）。 这种方法不仅强调感受野滑块内不同特征的重要性，而且优先考虑感受野空间特征。 通过这种方法，彻底解决了卷积核参数共享的问题。 感受野空间特征是根据卷积核的大小动态生成的，因此，RFA是卷积的固定组合，离不开卷积运算的帮助，同时依靠RFA来提高性能，所以我们 提出感受野注意卷积（RFAConv）。 具有3×3尺寸卷积核的RFAConv的整体结构如图2所示。

 作者设计了一种新的CBAM和CA，称为RFACBAM和RFACA，它专注于感受野的空间特征。与RFA类似，使用stride为k的k×k的最终卷积运算来提取特征信息，具体结构如图4和图5所示，将这2种新的卷积方法称为RFCBAMConv和RFCAConv。比较原始的CBAM，使用SE注意力来代替RFCBAM中的CAM。因为这样可以减少计算开销。

  9. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  10.     # Pad to 'same' shape outputs
  11.     if d > 1:
  12.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  13.     if p is None:
  14.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  15.     return p

  70. class MixConv2d(nn.Module):
  71.     # Mixed Depth-wise Conv https://arxiv.org/abs/1907.09595
  72.     def __init__(self, c1, c2, k=(1, 3), s=1, equal_ch=True):  # ch_in, ch_out, kernel, stride, ch_strategy
  73.         super().__init__()
  74.         n = len(k)  # number of convolutions
  75.         if equal_ch:  # equal c_ per group
  76.             i = torch.linspace(0, n - 1E-6, c2).floor()  # c2 indices
  77.             c_ = [(i == g).sum() for g in range(n)]  # intermediate channels
  78.         else:  # equal weight.numel() per group
  79.             b = [c2] + [0] * n
  80.             a = np.eye(n + 1, n, k=-1)
  81.             a -= np.roll(a, 1, axis=1)
  82.             a *= np.array(k) ** 2
  83.             a[0] = 1
  84.             c_ = np.linalg.lstsq(a, b, rcond=None)[0].round()  # solve for equal weight indices, ax = b
  85.  
  86.         self.m = nn.ModuleList([
  87.             nn.Conv2d(c1, int(c_), k, s, k // 2, groups=math.gcd(c1, int(c_)), bias=False) for k, c_ in zip(k, c_)])
  88.         self.bn = nn.BatchNorm2d(c2)
  89.         self.act = nn.SiLU()
  90.  
  91.     def forward(self, x):
  92.         return self.act(self.bn(torch.cat([m(x) for m in self.m], 1)))

  95. class CAConv(nn.Module):
  96.     def __init__(self, inp, oup, kernel_size, stride, reduction=32):
  97.         super(CAConv, self).__init__()
  98.         self.pool_h = nn.AdaptiveAvgPool2d((None, 1))
  99.         self.pool_w = nn.AdaptiveAvgPool2d((1, None))
  100.  
  101.         mip = max(8, inp // reduction)
  102.  
  103.         self.conv1 = nn.Conv2d(inp, mip, kernel_size=1, stride=1, padding=0)
  104.         self.bn1 = nn.BatchNorm2d(mip)
  105.         self.act = h_swish()
  106.  
  107.         self.conv_h = nn.Conv2d(mip, inp, kernel_size=1, stride=1, padding=0)
  108.         self.conv_w = nn.Conv2d(mip, inp, kernel_size=1, stride=1, padding=0)
  109.         self.conv = nn.Sequential(nn.Conv2d(inp, oup, kernel_size, padding=kernel_size // 2, stride=stride),
  110.                                   nn.BatchNorm2d(oup),
  111.                                   nn.ReLU())
  112.  
  113.     def forward(self, x):
  114.         identity = x
  115.  
  116.         n, c, h, w = x.size()
  117.         x_h = self.pool_h(x)
  118.         x_w = self.pool_w(x).permute(0, 1, 3, 2)
  119.  
  120.         y = torch.cat([x_h, x_w], dim=2)
  121.         y = self.conv1(y)
  122.         y = self.bn1(y)
  123.         y = self.act(y)
  124.  
  125.         x_h, x_w = torch.split(y, [h, w], dim=2)
  126.         x_w = x_w.permute(0, 1, 3, 2)
  127.  
  128.         a_h = self.conv_h(x_h).sigmoid()
  129.         a_w = self.conv_w(x_w).sigmoid()
  130.  
  131.         out = identity * a_w * a_h
  132.  
  133.         return self.conv(out)
  134. class CBAMConv(nn.Module):
  135.     def __init__(self, channel, out_channel, kernel_size, stride, reduction=16, spatial_kernel=7):
  136.         super().__init__()
  137.  
  138.         self.max_pool = nn.AdaptiveMaxPool2d(1)
  139.         self.avg_pool = nn.AdaptiveAvgPool2d(1)
  140.         self.mlp = nn.Sequential(
  141.             nn.Conv2d(channel, channel // reduction, 1, bias=False),
  142.             nn.ReLU(inplace=True),
  143.             nn.Conv2d(channel // reduction, channel, 1, bias=False)
  144.         )
  145.  
  146.         self.spatital = nn.Conv2d(2, 1, kernel_size=spatial_kernel,
  147.                                   padding=spatial_kernel // 2, bias=False)
  148.         self.sigmoid = nn.Sigmoid()
  149.  
  150.         self.conv = nn.Sequential(nn.Conv2d(channel, out_channel, kernel_size, padding=kernel_size // 2, stride=stride),
  151.                                   nn.BatchNorm2d(out_channel),
  152.                                   nn.ReLU())
  153.         kernel_sizes = [3, 5, 7]
  154.         self.attention_layers = nn.ModuleList([
  155.             nn.Sequential(
  156.                 nn.Conv2d(2, 1, kernel_size=k, padding=k // 2),
  157.                 nn.Sigmoid()
  158.             ) for k in kernel_sizes
  159.         ])
  160.  
  161.     def forward(self, x):
  162.         max_out = self.mlp(self.max_pool(x))
  163.         avg_out = self.mlp(self.avg_pool(x))
  164.         channel_out = self.sigmoid(max_out + avg_out)
  165.         x = channel_out * x
  166.  
  167.         max_out, _ = torch.max(x, dim=1, keepdim=True)
  168.         avg_out = torch.mean(x, dim=1, keepdim=True)
  169.         spatial_out = self.sigmoid(self.spatital(torch.cat([max_out, avg_out], dim=1)))
  170.         spatial_out = [att_layer(torch.cat([max_out, avg_out], dim=1)) for att_layer in self.attention_layers]
  171.         spatial_out = self.sigmoid(sum(spatial_out)) #/ len(spatial_out)
  172.         x = spatial_out * x
  173.         return self.conv(x)

  242. class RFAConv(nn.Module):
  243.     def __init__(self, in_channels, out_channels, kernel_size, stride=1):
  244.         super(RFAConv, self).__init__()
  245.  
  246.         # Adaptive Average Pooling
  247.         self.avg_pool = nn.AdaptiveAvgPool2d(1)
  248.  
  249.         # Non-shared Convolution layers
  250.         self.conv1 = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=kernel_size//2, groups=in_channels)
  251.         self.conv2 = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=kernel_size//2, groups=in_channels)
  252.         self.conv3 = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=kernel_size//2, groups=in_channels)
  253.         self.bn = nn.BatchNorm2d(out_channels)
  254.         self.act = nn.SiLU
  255.         self.softmax = nn.Softmax(dim=1)
  256.         self.conv4 = nn.Conv2d(in_channels, out_channels, 1, 1)
  257.  
  258.     def forward(self, x):
  259.         b, c, _, _ = x.size()
  260.  
  261.         y = self.avg_pool(x)
  262.         y = y.view(b, c)
  263.  
  264.         w1 = self.softmax(y)
  265.         w2 = self.softmax(y)
  266.         w3 = self.softmax(y)
  267.  
  268.         x1 = self.conv1(x)
  269.         x2 = self.conv2(x)
  270.         x3 = self.conv3(x)
  271.         if w1.shape[1] == x1.shape[1]:
  272.             out = w1.unsqueeze(2).unsqueeze(3) * x1 + \
  273.                   w2.unsqueeze(2).unsqueeze(3) * x2 + \
  274.                   w3.unsqueeze(2).unsqueeze(3) * x3
  275.         else:
  276.             out = self.conv4(w1.unsqueeze(2).unsqueeze(3)) * x1+ \
  277.                 self.conv4(w2.unsqueeze(2).unsqueeze(3)) * x2+ \
  278.                 self.conv4(w3.unsqueeze(2).unsqueeze(3)) * x3
  279.  
  280.         return out

  520. class SKConv(nn.Module):
  521.  
  522.     def __init__(self, channel=512, c2=512 ,kernel_size=3,stride=2, kernels=[1,3,5,7],reduction=32,group=16,L=32):
  523.         super().__init__()
  524.         inp =channel
  525.         group = channel
  526.         oup = c2
  527.         self.d=max(L,channel//reduction)
  528.         self.convs=nn.ModuleList([])
  529.         for k in kernels:
  530.             self.convs.append(
  531.                 nn.Sequential(collections.OrderedDict([
  532.                     ('conv',nn.Conv2d(channel,channel,kernel_size=k,padding=k//2,groups=group)),
  533.                     ('bn',nn.BatchNorm2d(channel)),
  534.                     ('relu',nn.ReLU()),
  535.                 ]))
  536.             )
  537.         self.fc=nn.Linear(channel * 21 ,self.d)
  538.         self.fcs=nn.ModuleList([])
  539.         for i in range(len(kernels)):
  540.             self.fcs.append(nn.Linear(self.d,channel))
  541.         self.softmax=nn.Softmax(dim=0)
  542.         self.avg_pool1 =    nn.AdaptiveAvgPool2d(1)
  543.         self.avg_pool2 =    nn.AdaptiveAvgPool2d(2)
  544.         self.avg_pool4 =    nn.AdaptiveAvgPool2d(4)
  545.         self.conv = nn.Sequential(nn.Conv2d(inp, oup, kernel_size, padding=kernel_size // 2, stride=stride),
  546.                                   nn.BatchNorm2d(oup),
  547.                                   nn.ReLU())
  548.     def forward(self, x):
  549.         bs, c, _, _ = x.size()
  550.         conv_outs=[]
  551.         ### split
  552.         for conv in self.convs:
  553.             conv_outs.append(conv(x))
  554.         feats=torch.stack(conv_outs,0)#k,bs,channel,h,w
  555.  
  556.         ### fuse
  557.         U=sum(conv_outs) #bs,c,h,w
  558.  
  559.         y1 = self.avg_pool1(x).reshape((bs, -1))
  560.         y2 = self.avg_pool2(x).reshape((bs, -1))
  561.         y3 = self.avg_pool4(x).reshape((bs, -1))
  562.         S = torch.cat((y1, y2, y3), 1)
  563.         ### reduction channel
  564.         # S=U.mean(-1).mean(-1) #bs,c
  565.         Z=self.fc(S) #bs,d
  566.  
  567.         ### calculate attention weight
  568.         weights=[]
  569.         for fc in self.fcs:
  570.             weight=fc(Z)
  571.             weights.append(weight.view(bs,c,1,1)) #bs,channel
  572.         attention_weughts=torch.stack(weights,0)#k,bs,channel,1,1
  573.         attention_weughts=self.softmax(attention_weughts)#k,bs,channel,1,1
  574.  
  575.         ### fuse
  576.         V=(attention_weughts*feats).sum(0)
  577.         V = self.conv(V)
  578.         return V

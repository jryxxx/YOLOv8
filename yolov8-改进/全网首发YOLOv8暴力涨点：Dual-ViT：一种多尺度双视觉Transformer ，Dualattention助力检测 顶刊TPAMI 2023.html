  100. class Attention(nn.Module):
  101.     def __init__(self, dim, num_heads):
  102.         super().__init__()
  103.         assert dim % num_heads == 0, f"dim {dim} should be divided by num_heads {num_heads}."
  104.  
  105.         self.dim = dim
  106.         self.num_heads = num_heads
  107.         head_dim = dim // num_heads
  108.         self.scale = head_dim ** -0.5
  109.  
  110.         self.q = nn.Linear(dim, dim)
  111.         self.kv = nn.Linear(dim, dim * 2)
  112.         self.proj = nn.Linear(dim, dim)
  113.         self.apply(self._init_weights)
  114.  
  115.     def _init_weights(self, m):
  116.         if isinstance(m, nn.Linear):
  117.             trunc_normal_(m.weight, std=.02)
  118.             if isinstance(m, nn.Linear) and m.bias is not None:
  119.                 nn.init.constant_(m.bias, 0)
  120.         elif isinstance(m, nn.LayerNorm):
  121.             if m.bias is not None:
  122.                 nn.init.constant_(m.bias, 0)
  123.             if m.weight is not None:
  124.                 nn.init.constant_(m.weight, 1.0)
  125.         elif isinstance(m, nn.Conv2d):
  126.             fan_out = m.kernel_size[0] * m.kernel_size[1] * m.out_channels
  127.             fan_out //= m.groups
  128.             m.weight.data.normal_(0, math.sqrt(2.0 / fan_out))
  129.             if m.bias is not None:
  130.                 m.bias.data.zero_()

  132.     def forward(self, x):
  133.        # x =x.permute(3, 0, 1, 2)
  134.         B, H, W, C = x.shape
  135.  
  136.         N = H * W
  137.         q = self.q(x).reshape(B, N, self.num_heads, C // self.num_heads).permute(0, 2, 1, 3)
  138.         kv = self.kv(x).reshape(B, -1, 2, self.num_heads, C // self.num_heads).permute(2, 0, 3, 1, 4)
  139.         k, v = kv[0], kv[1]

  142.         x = (attn @ v).transpose(1, 2).reshape(B, H, W , C)
  143.         x = self.proj(x)

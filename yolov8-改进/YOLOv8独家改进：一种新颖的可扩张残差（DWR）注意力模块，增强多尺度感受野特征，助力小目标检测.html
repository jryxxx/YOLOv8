  2. from torch import nn

  4. def autopad(k, p=None, d=1):  # kernel, padding, dilation
  5.     # Pad to 'same' shape outputs
  6.     if d > 1:
  7.         k = d * (k - 1) + 1 if isinstance(k, int) else [d * (x - 1) + 1 for x in k]  # actual kernel-size
  8.     if p is None:
  9.         p = k // 2 if isinstance(k, int) else [x // 2 for x in k]  # auto-pad
  10.     return p
  11.  
  12.  
  13. class Conv(nn.Module):
  14.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  15.     default_act = nn.SiLU()  # default activation
  16.  
  17.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  18.         super().__init__()
  19.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  20.         self.bn = nn.BatchNorm2d(c2)
  21.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  22.  
  23.     def forward(self, x):
  24.         return self.act(self.bn(self.conv(x)))
  25.  
  26.     def forward_fuse(self, x):
  27.         return self.act(self.conv(x))
  28.         
  29. class C2f(nn.Module):
  30.     # CSP Bottleneck with 2 convolutions
  31.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  32.         super().__init__()
  33.         self.c = int(c2 * e)  # hidden channels
  34.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  35.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)
  36.         self.m = nn.ModuleList(Bottleneck(self.c, self.c, shortcut, g, k=((3, 3), (3, 3)), e=1.0) for _ in range(n))
  37.  
  38.     def forward(self, x):
  39.         y = list(self.cv1(x).chunk(2, 1))
  40.         y.extend(m(y[-1]) for m in self.m)
  41.         return self.cv2(torch.cat(y, 1))
  42.  
  43.     def forward_split(self, x):
  44.         y = list(self.cv1(x).split((self.c, self.c), 1))
  45.         y.extend(m(y[-1]) for m in self.m)
  46.         return self.cv2(torch.cat(y, 1))
  47.  
  48. class Bottleneck(nn.Module):
  49.     # Standard bottleneck
  50.     def __init__(self, c1, c2, shortcut=True, g=1, k=(3, 3), e=0.5):  # ch_in, ch_out, shortcut, groups, kernels, expand
  51.         super().__init__()
  52.         c_ = int(c2 * e)  # hidden channels
  53.         self.cv1 = Conv(c1, c_, k[0], 1)
  54.         self.cv2 = Conv(c_, c2, k[1], 1, g=g)
  55.         self.add = shortcut and c1 == c2
  56.  
  57.     def forward(self, x):
  58.         return x + self.cv2(self.cv1(x)) if self.add else self.cv2(self.cv1(x))
  59.  
  60.  
  61. class DWR(nn.Module):
  62.     def __init__(self, dim) -> None:
  63.         super().__init__()
  64.  
  65.         self.conv_3x3 = Conv(dim, dim // 2, 3)
  66.         
  67.         self.conv_3x3_d1 = Conv(dim // 2, dim, 3, d=1)
  68.         self.conv_3x3_d3 = Conv(dim // 2, dim // 2, 3, d=3)
  69.         self.conv_3x3_d5 = Conv(dim // 2, dim // 2, 3, d=5)
  70.         
  71.         self.conv_1x1 = Conv(dim * 2, dim, k=1)
  72.         
  73.     def forward(self, x):
  74.         conv_3x3 = self.conv_3x3(x)
  75.         x1, x2, x3 = self.conv_3x3_d1(conv_3x3), self.conv_3x3_d3(conv_3x3), self.conv_3x3_d5(conv_3x3)
  76.         x_out = torch.cat([x1, x2, x3], dim=1)
  77.         x_out = self.conv_1x1(x_out) + x
  78.         return x_out
  79.  
  80.  
  81. class C2f_DWR(C2f):
  82.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):
  83.         super().__init__(c1, c2, n, shortcut, g, e)
  84.         self.m = nn.ModuleList(DWR(self.c) for _ in range(n))

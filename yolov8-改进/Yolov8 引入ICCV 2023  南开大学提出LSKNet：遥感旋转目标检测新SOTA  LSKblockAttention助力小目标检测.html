  6. class LSKblock(nn.Module):
  7.     def __init__(self, dim):
  8.         super().__init__()
  9.         self.conv0 = nn.Conv2d(dim, dim, 5, padding=2, groups=dim)
  10.         self.conv_spatial = nn.Conv2d(dim, dim, 7, stride=1, padding=9, groups=dim, dilation=3)
  11.         self.conv1 = nn.Conv2d(dim, dim//2, 1)
  12.         self.conv2 = nn.Conv2d(dim, dim//2, 1)
  13.         self.conv_squeeze = nn.Conv2d(2, 2, 7, padding=3)
  14.         self.conv = nn.Conv2d(dim//2, dim, 1)
  15.  
  16.     def forward(self, x):   
  17.         attn1 = self.conv0(x)
  18.         attn2 = self.conv_spatial(attn1)
  19.  
  20.         attn1 = self.conv1(attn1)
  21.         attn2 = self.conv2(attn2)
  22.         
  23.         attn = torch.cat([attn1, attn2], dim=1)
  24.         avg_attn = torch.mean(attn, dim=1, keepdim=True)
  25.         max_attn, _ = torch.max(attn, dim=1, keepdim=True)
  26.         agg = torch.cat([avg_attn, max_attn], dim=1)
  27.         sig = self.conv_squeeze(agg).sigmoid()
  28.         attn = attn1 * sig[:,0,:,:].unsqueeze(1) + attn2 * sig[:,1,:,:].unsqueeze(1)
  29.         attn = self.conv(attn)
  30.         return x * attn
  31.  
  32.

  34. class LSKblockAttention(nn.Module):
  35.     def __init__(self, d_model):
  36.         super().__init__()
  37.  
  38.         self.proj_1 = nn.Conv2d(d_model, d_model, 1)
  39.         self.activation = nn.GELU()
  40.         self.spatial_gating_unit = LSKblock(d_model)
  41.         self.proj_2 = nn.Conv2d(d_model, d_model, 1)
  42.  
  43.     def forward(self, x):
  44.         shorcut = x.clone()
  45.         x = self.proj_1(x)
  46.         x = self.activation(x)
  47.         x = self.spatial_gating_unit(x)
  48.         x = self.proj_2(x)
  49.         x = x + shorcut
  50.         return x

  162.     # Pad to 'same' shape outputs

  170. class Conv(nn.Module):
  171.     # Standard convolution with args(ch_in, ch_out, kernel, stride, padding, groups, dilation, activation)
  172.     default_act = nn.SiLU()  # default activation
  173.  
  174.     def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
  175.         super().__init__()
  176.         self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
  177.         self.bn = nn.BatchNorm2d(c2)
  178.         self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()
  179.  
  180.     def forward(self, x):
  181.         return self.act(self.bn(self.conv(x)))

  184.         return self.act(self.conv(x))

  186. class C2f(nn.Module):
  187.     # CSP Bottleneck with 2 convolutions
  188.     def __init__(self, c1, c2, n=1, shortcut=False, g=1, e=0.5):  # ch_in, ch_out, number, shortcut, groups, expansion
  189.         super().__init__()
  190.         self.c = int(c2 * e)  # hidden channels
  191.         self.cv1 = Conv(c1, 2 * self.c, 1, 1)
  192.         self.cv2 = Conv((2 + n) * self.c, c2, 1)  # optional act=FReLU(c2)

  205. class Bottleneck(nn.Module):
  206.     # Standard bottleneck
  207.     def __init__(self, c1, c2, shortcut=True, g=1, k=(3, 3), e=0.5):  # ch_in, ch_out, shortcut, groups, kernels, expand
  208.         super().__init__()
  209.         c_ = int(c2 * e)  # hidden channels
  210.         self.cv1 = Conv(c1, c_, k[0], 1)
  211.         self.cv2 = Conv(c_, c2, k[1], 1, g=g)
  212.         self.add = shortcut and c1 == c2
  213.  
  214.     def forward(self, x):
  215.         return x + self.cv2(self.cv1(x)) if self.add else self.cv2(self.cv1(x))
